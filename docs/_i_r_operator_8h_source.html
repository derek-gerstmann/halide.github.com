109 /** Coerce the two expressions to have the same type, using C-style
110  * casting rules. For the purposes of casting, a boolean type is
111  * UInt(1). We use the following procedure:
112  *
113  * If the types already match, do nothing.
114  *
115  * Then, if one type is a vector and the other is a scalar, the scalar
116  * is broadcast to match the vector width, and we continue.
117  *
118  * Then, if one type is floating-point and the other is not, the
119  * non-float is cast to the floating-point type, and we're done.
120  *
121  * Then, if neither is a float but one of the two is a constant, the
122  * constant is cast to match the non-const type and we're done. For
123  * example, e has type UInt(8), then (e*32) also has type UInt(8),
124  * despite the overflow that may occur. Note that this also means that
125  * (e*(-1)) is positive, and is equivalent to (e*255) - i.e. the (-1)
126  * is cast to a UInt(8) before the multiplication.
127  *
128  * Then, if both types are unsigned ints, the one with fewer bits is
129  * cast to match the one with more bits and we're done.
130  *
131  * Then, if both types are signed ints, the one with fewer bits is
132  * cast to match the one with more bits and we're done.
133  *
134  * Finally, if one type is an unsigned int and the other type is a signed
135  * int, both are cast to a signed int with the greater of the two
136  * bit-widths. For example, matching an Int(8) with a UInt(16) results
137  * in an Int(16).
138  *
139  */

1062 /** Linear interpolate between the two values according to a weight.
1063  * \param zero_val The result when weight is 0
1064  * \param one_val The result when weight is 1
1065  * \param weight The interpolation amount
1066  *
1067  * Both zero_val and one_val must have the same type. All types are
1068  * supported, including bool.
1069  *
1070  * The weight is treated as its own type and must be float or an
1071  * unsigned integer type. It is scaled to the bit-size of the type of
1072  * x and y if they are integer, or converted to float if they are
1073  * float. Integer weights are converted to float via division by the
1074  * full-range value of the weight's type. Floating-point weights used
1075  * to interpolate between integer values must be between 0.0f and
1076  * 1.0f, and an error may be signaled if it is not provably so. (clamp
1077  * operators can be added to provide proof. Currently an error is only
1078  * signalled for constant weights.)
1079  *
1080  * For integer linear interpolation, out of range values cannot be
1081  * represented. In particular, weights that are conceptually less than
1082  * 0 or greater than 1.0 are not representable. As such the result is
1083  * always between x and y (inclusive of course). For lerp with
1084  * floating-point values and floating-point weight, the full range of
1085  * a float is valid, however underflow and overflow can still occur.
1086  *
1087  * Ordering is not required between zero_val and one_val:
1088  * lerp(42, 69, .5f) == lerp(69, 42, .5f) == 56
1089  *
1090  * Results for integer types are for exactly rounded arithmetic. As
1091  * such, there are cases where 16-bit and float differ because 32-bit
1092  * floating-point (float) does not have enough precision to produce
1093  * the exact result. (Likely true for 32-bit integer
1094  * vs. double-precision floating-point as well.)
1095  *
1096  * At present, double precision and 64-bit integers are not supported.
1097  *
1098  * Generally, lerp will vectorize as if it were an operation on a type
1099  * twice the bit size of the inferred type for x and y.
1100  *
1101  * Some examples:
1102  * \code
1103  *
1104  * // Since Halide does not have direct type delcarations, casts
1105  * // below are used to indicate the types of the parameters.
1106  * // Such casts not required or expected in actual code where types
1107  * // are inferred.
1108  *
1109  * lerp(cast<float>(x), cast<float>(y), cast<float>(w)) ->
1110  * x * (1.0f - w) + y * w
1111  *
1112  * lerp(cast<uint8_t>(x), cast<uint8_t>(y), cast<uint8_t>(w)) ->
1113  * cast<uint8_t>(cast<uint8_t>(x) * (1.0f - cast<uint8_t>(w) / 255.0f) +
1114  * cast<uint8_t>(y) * cast<uint8_t>(w) / 255.0f + .5f)
1115  *
1116  * // Note addition in Halide promoted uint8_t + int8_t to int16_t already,
1117  * // the outer cast is added for clarity.
1118  * lerp(cast<uint8_t>(x), cast<int8_t>(y), cast<uint8_t>(w)) ->
1119  * cast<int16_t>(cast<uint8_t>(x) * (1.0f - cast<uint8_t>(w) / 255.0f) +
1120  * cast<int8_t>(y) * cast<uint8_t>(w) / 255.0f + .5f)
1121  *
1122  * lerp(cast<int8_t>(x), cast<int8_t>(y), cast<float>(w)) ->
1123  * cast<int8_t>(cast<int8_t>(x) * (1.0f - cast<float>(w)) +
1124  * cast<int8_t>(y) * cast<uint8_t>(w))
1125  *
1126  * \endcode
1127  * */
1128 inline Expr lerp(Expr zero_val, Expr one_val, Expr weight) {
1129  user_assert(zero_val.defined()) << "lerp with undefined zero value";
1130  user_assert(one_val.defined()) << "lerp with undefined one value";
1131  user_assert(weight.defined()) << "lerp with undefined weight";
1132 
1133  // We allow integer constants through, so that you can say things
1134  // like lerp(0, cast<uint8_t>(x), alpha) and produce an 8-bit
1135  // result. Note that lerp(0.0f, cast<uint8_t>(x), alpha) will
1136  // produce an error, as will lerp(0.0f, cast<double>(x),
1137  // alpha). lerp(0, cast<float>(x), alpha) is also allowed and will
1138  // produce a float result.
1139  if (as_const_int(zero_val)) {
1140  zero_val = cast(one_val.type(), zero_val);
1141  }
1142  if (as_const_int(one_val)) {
1143  one_val = cast(zero_val.type(), one_val);
1144  }
1145 
1146  user_assert(zero_val.type() == one_val.type())
1147  << "Can't lerp between " << zero_val << " of type " << zero_val.type()
1148  << " and " << one_val << " of different type " << one_val.type() << "\n";
1149  user_assert((weight.type().is_uint() || weight.type().is_float()))
1150  << "A lerp weight must be an unsigned integer or a float, but "
1151  << "lerp weight " << weight << " has type " << weight.type() << ".\n";
1152  user_assert((zero_val.type().is_float() || zero_val.type().width <= 32))
1153  << "Lerping between 64-bit integers is not supported\n";
1154  // Compilation error for constant weight that is out of range for integer use
1155  // as this seems like an easy to catch gotcha.
1156  if (!zero_val.type().is_float()) {
1157  const float *const_weight = as_const_float(weight);
1158  if (const_weight) {
1159  user_assert(*const_weight >= 0.0f && *const_weight <= 1.0f)
1160  << "Floating-point weight for lerp with integer arguments is "
1161  << *const_weight << ", which is not in the range [0.0f, 1.0f].\n";
1162  }
1163  }
1164  return Internal::Call::make(zero_val.type(), Internal::Call::lerp,
1165  {zero_val, one_val, weight},
1166  Internal::Call::Intrinsic);
1167 }
