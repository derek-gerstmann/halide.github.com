00096 /** Coerce the two expressions to have the same type, using C-style
00097  * casting rules. For the purposes of casting, a boolean type is
00098  * UInt(1). We use the following procedure:
00099  *
00100  * If the types already match, do nothing.
00101  *
00102  * Then, if one type is a vector and the other is a scalar, the scalar
00103  * is broadcast to match the vector width, and we continue.
00104  *
00105  * Then, if one type is floating-point and the other is not, the
00106  * non-float is cast to the floating-point type, and we're done.
00107  *
00108  * Then, if neither is a float but one of the two is a constant, the
00109  * constant is cast to match the non-const type and we're done. For
00110  * example, e has type UInt(8), then (e*32) also has type UInt(8),
00111  * despite the overflow that may occur. Note that this also means that
00112  * (e*(-1)) is positive, and is equivalent to (e*255) - i.e. the (-1)
00113  * is cast to a UInt(8) before the multiplication.
00114  *
00115  * Then, if both types are unsigned ints, the one with fewer bits is
00116  * cast to match the one with more bits and we're done.
00117  *
00118  * Then, if both types are signed ints, the one with fewer bits is
00119  * cast to match the one with more bits and we're done.
00120  *
00121  * Finally, if one type is an unsigned int and the other type is a signed
00122  * int, both are cast to a signed int with the greater of the two
00123  * bit-widths. For example, matching an Int(8) with a UInt(16) results
00124  * in an Int(16).
00125  *
00126  */
00127 EXPORT void match_types(Expr &a, Expr &b);

00884 /** Linear interpolate between the two values according to a weight.
00885  * \param zero_val The result when weight is 0
00886  * \param one_val The result when weight is 1
00887  * \param weight The interpolation amount
00888  *
00889  * Both zero_val and one_val must have the same type. All types are
00890  * supported, including bool.
00891  *
00892  * The weight is treated as its own type and must be float or an
00893  * unsigned integer type. It is scaled to the bit-size of the type of
00894  * x and y if they are integer, or converted to float if they are
00895  * float. Integer weights are converted to float via division by the
00896  * full-range value of the weight's type. Floating-point weights used
00897  * to interpolate between integer values must be between 0.0f and
00898  * 1.0f, and an error may be signaled if it is not provably so. (clamp
00899  * operators can be added to provide proof. Currently an error is only
00900  * signalled for constant weights.)
00901  *
00902  * For integer linear interpolation, out of range values cannot be
00903  * represented. In particular, weights that are conceptually less than
00904  * 0 or greater than 1.0 are not representable. As such the result is
00905  * always between x and y (inclusive of course). For lerp with
00906  * floating-point values and floating-point weight, the full range of
00907  * a float is valid, however underflow and overflow can still occur.
00908  *
00909  * Ordering is not required between zero_val and one_val:
00910  *     lerp(42, 69, .5f) == lerp(69, 42, .5f) == 56
00911  *
00912  * Results for integer types are for exactly rounded arithmetic. As
00913  * such, there are cases where 16-bit and float differ because 32-bit
00914  * floating-point (float) does not have enough precision to produce
00915  * the exact result. (Likely true for 32-bit integer
00916  * vs. double-precision floating-point as well.)
00917  *
00918  * At present, double precision and 64-bit integers are not supported.
00919  *
00920  * Generally, lerp will vectorize as if it were an operation on a type
00921  * twice the bit size of the inferred type for x and y.
00922  *
00923  * Some examples:
00924  * \code
00925  *
00926  *     // Since Halide does not have direct type delcarations, casts
00927  *     // below are used to indicate the types of the parameters.
00928  *     // Such casts not required or expected in actual code where types
00929  *     // are inferred.
00930  *
00931  *     lerp(cast<float>(x), cast<float>(y), cast<float>(w)) ->
00932  *       x * (1.0f - w) + y * w
00933  *
00934  *     lerp(cast<uint8_t>(x), cast<uint8_t>(y), cast<uint8_t>(w)) ->
00935  *       cast<uint8_t>(cast<uint8_t>(x) * (1.0f - cast<uint8_t>(w) / 255.0f) +
00936  *                     cast<uint8_t>(y) * cast<uint8_t>(w) / 255.0f + .5f)
00937  *
00938  *     // Note addition in Halide promoted uint8_t + int8_t to int16_t already,
00939  *     // the outer cast is added for clarity.
00940  *     lerp(cast<uint8_t>(x), cast<int8_t>(y), cast<uint8_t>(w)) ->
00941  *       cast<int16_t>(cast<uint8_t>(x) * (1.0f - cast<uint8_t>(w) / 255.0f) +
00942  *                     cast<int8_t>(y) * cast<uint8_t>(w) / 255.0f + .5f)
00943  *
00944  *     lerp(cast<int8_t>(x), cast<int8_t>(y), cast<float>(w)) ->
00945  *       cast<int8_t>(cast<int8_t>(x) * (1.0f - cast<float>(w)) +
00946  *                    cast<int8_t>(y) * cast<uint8_t>(w))
00947  *
00948  * \endcode
00949  * */
00950 inline Expr lerp(Expr zero_val, Expr one_val, Expr weight) {
00951     user_assert(zero_val.defined()) << "lerp with undefined zero value";
00952     user_assert(one_val.defined()) << "lerp with undefined one value";
00953     user_assert(weight.defined()) << "lerp with undefined weight";
00954 
00955     // We allow integer constants through, so that you can say things
00956     // like lerp(0, cast<uint8_t>(x), alpha) and produce an 8-bit
00957     // result. Note that lerp(0.0f, cast<uint8_t>(x), alpha) will
00958     // produce an error, as will lerp(0.0f, cast<double>(x),
00959     // alpha). lerp(0, cast<float>(x), alpha) is also allowed and will
00960     // produce a float result.
00961     if (as_const_int(zero_val)) {
00962         zero_val = cast(one_val.type(), zero_val);
00963     }
00964     if (as_const_int(one_val)) {
00965         one_val = cast(zero_val.type(), one_val);
00966     }
00967 
00968     user_assert(zero_val.type() == one_val.type())
00969         << "Can't lerp between " << zero_val << " of type " << zero_val.type()
00970         << " and " << one_val << " of different type " << one_val.type() << "\n";
00971     user_assert((weight.type().is_uint() || weight.type().is_float()))
00972         << "A lerp weight must be an unsigned integer or a float, but "
00973         << "lerp weight " << weight << " has type " << weight.type() << ".\n";
00974     user_assert((zero_val.type().is_float() || zero_val.type().width <= 32))
00975         << "Lerping between 64-bit integers is not supported\n";
00976     // Compilation error for constant weight that is out of range for integer use
00977     // as this seems like an easy to catch gotcha.
00978     if (!zero_val.type().is_float()) {
00979         const float *const_weight = as_const_float(weight);
00980         if (const_weight) {
00981             user_assert(*const_weight >= 0.0f && *const_weight <= 1.0f)
00982                 << "Floating-point weight for lerp with integer arguments is "
00983                 << *const_weight << ", which is not in the range [0.0f, 1.0f].\n";
00984         }
00985     }
00986     return Internal::Call::make(zero_val.type(), Internal::Call::lerp,
00987                                 vec(zero_val, one_val, weight),
00988                                 Internal::Call::Intrinsic);
00989 }

01014 /** Return a random variable representing a uniformly distributed
01015  * float in the half-open interval [0.0f, 1.0f). For random numbers of
01016  * other types, use lerp with a random float as the last parameter.
01017  *
01018  * Optionally takes a seed.
01019  *
01020  * Note that:
01021  \code
01022  Expr x = random_float();
01023  Expr y = x + x;
01024  \endcode
01025  *
01026  * is very different to
01027  *
01028  \code

01032  * The first doubles a random variable, and the second adds two

01152 /** Return an undef value of the given type. Halide skips stores that
01153  * depend on undef values, so you can use this to mean "do not modify
01154  * this memory location". This is an escape hatch that can be used for
01155  * several things:
01156  *
01157  * You can define a reduction with no pure step, by setting the pure
01158  * step to undef. Do this only if you're confident that the update
01159  * steps are sufficient to correctly fill in the domain.
01160  *
01161  * For a tuple-valued reduction, you can write an update step that
01162  * only updates some tuple elements.
01163  *
01164  * You can define single-stage pipeline that only has update steps,
01165  * and depends on the values already in the output buffer.
01166  *
01167  * Use this feature with great caution, as you can use it to load from
01168  * uninitialized memory.
01169  */
01170 inline Expr undef(Type t) {
01171     return Internal::Call::make(t, Internal::Call::undef,
01172                                 std::vector<Expr>(),
01173                                 Internal::Call::Intrinsic);
01174 }
