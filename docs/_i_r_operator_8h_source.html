130 /** Coerce the two expressions to have the same type, using C-style
131  * casting rules. For the purposes of casting, a boolean type is
132  * UInt(1). We use the following procedure:
133  *
134  * If the types already match, do nothing.
135  *
136  * Then, if one type is a vector and the other is a scalar, the scalar
137  * is broadcast to match the vector width, and we continue.
138  *
139  * Then, if one type is floating-point and the other is not, the
140  * non-float is cast to the floating-point type, and we're done.
141  *
142  * Then, if both types are unsigned ints, the one with fewer bits is
143  * cast to match the one with more bits and we're done.
144  *
145  * Then, if both types are signed ints, the one with fewer bits is
146  * cast to match the one with more bits and we're done.
147  *
148  * Finally, if one type is an unsigned int and the other type is a signed
149  * int, both are cast to a signed int with the greater of the two
150  * bit-widths. For example, matching an Int(8) with a UInt(16) results
151  * in an Int(16).
152  *
153  */

828

839  false_value = cast(true_value.type(), false_value);

1453 /** Linear interpolate between the two values according to a weight.
1454  * \param zero_val The result when weight is 0
1455  * \param one_val The result when weight is 1
1456  * \param weight The interpolation amount
1457  *
1458  * Both zero_val and one_val must have the same type. All types are
1459  * supported, including bool.
1460  *
1461  * The weight is treated as its own type and must be float or an
1462  * unsigned integer type. It is scaled to the bit-size of the type of
1463  * x and y if they are integer, or converted to float if they are
1464  * float. Integer weights are converted to float via division by the
1465  * full-range value of the weight's type. Floating-point weights used
1466  * to interpolate between integer values must be between 0.0f and
1467  * 1.0f, and an error may be signaled if it is not provably so. (clamp
1468  * operators can be added to provide proof. Currently an error is only
1469  * signalled for constant weights.)
1470  *
1471  * For integer linear interpolation, out of range values cannot be
1472  * represented. In particular, weights that are conceptually less than
1473  * 0 or greater than 1.0 are not representable. As such the result is
1474  * always between x and y (inclusive of course). For lerp with
1475  * floating-point values and floating-point weight, the full range of
1476  * a float is valid, however underflow and overflow can still occur.
1477  *
1478  * Ordering is not required between zero_val and one_val:
1479  * lerp(42, 69, .5f) == lerp(69, 42, .5f) == 56
1480  *
1481  * Results for integer types are for exactly rounded arithmetic. As
1482  * such, there are cases where 16-bit and float differ because 32-bit
1483  * floating-point (float) does not have enough precision to produce
1484  * the exact result. (Likely true for 32-bit integer
1485  * vs. double-precision floating-point as well.)
1486  *
1487  * At present, double precision and 64-bit integers are not supported.
1488  *
1489  * Generally, lerp will vectorize as if it were an operation on a type
1490  * twice the bit size of the inferred type for x and y.
1491  *
1492  * Some examples:
1493  * \code
1494  *
1495  * // Since Halide does not have direct type delcarations, casts
1496  * // below are used to indicate the types of the parameters.
1497  * // Such casts not required or expected in actual code where types
1498  * // are inferred.
1499  *
1500  * lerp(cast<float>(x), cast<float>(y), cast<float>(w)) ->
1501  * x * (1.0f - w) + y * w
1502  *

1515  * cast<int8_t>(y) * cast<uint8_t>(w))

1518  * */

1521  user_assert(one_val.defined()) << "lerp with undefined one value";
1522  user_assert(weight.defined()) << "lerp with undefined weight";

1525  // like lerp(0, cast<uint8_t>(x), alpha) and produce an 8-bit
1526  // result. Note that lerp(0.0f, cast<uint8_t>(x), alpha) will
1527  // produce an error, as will lerp(0.0f, cast<double>(x),
1528  // alpha). lerp(0, cast<float>(x), alpha) is also allowed and will
1529  // produce a float result.
1530  if (as_const_int(zero_val)) {
1531  zero_val = cast(one_val.type(), zero_val);
1532  }
1533  if (as_const_int(one_val)) {
1534  one_val = cast(zero_val.type(), one_val);
1535  }
1536 
1537  user_assert(zero_val.type() == one_val.type())

1546  // as this seems like an easy to catch gotcha.
1547  if (!zero_val.type().is_float()) {

1549  if (const_weight) {
1550  user_assert(*const_weight >= 0.0 && *const_weight <= 1.0)

1552  << *const_weight << ", which is not in the range [0.0, 1.0].\n";
1553  }
1554  }

1556  {zero_val, one_val, weight},
