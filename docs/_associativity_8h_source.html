19 /**
20  * Represent the equivalent associative op of an update definition.
21  * For example, the following associative Expr, min(f(x), g(r.x) + 2),
22  * where f(x) is the self-recurrence term, is represented as:
23  \code
24  AssociativeOp assoc(
25  AssociativePattern(min(x, y), +inf, true),
26  {Replacement("x", f(x))},
27  {Replacement("y", g(r.x) + 2)},
28  true
29  );
30  \endcode
31  *
32  * 'pattern' contains the list of equivalent binary/unary operators (+ identities)
33  * for each Tuple element in the update definition. 'pattern' also contains
34  * a boolean that indicates if the op is also commutative. 'xs' and 'ys'
35  * contain the corresponding definition of each variable in the list of
36  * binary operators.
37  *
38  * For unary operator, 'xs' is not set, i.e. it will be a pair of empty string
39  * and undefined Expr: {"", Expr()}. 'pattern' will only contain the 'y' term in
40  * this case. For example, min(g(r.x), 4), will be represented as:
41  \code
42  AssociativeOp assoc(
43  AssociativePattern(y, 0, false),
44  {Replacement("", Expr())},
45  {Replacement("y", min(g(r.x), 4))},
46  true
47  );
48  \endcode
49  * Since it is a unary operator, the identity does not matter. It can be
50  * anything.
51  */
