24  // Worker threads are divided into an 'A' team and a 'B' team. The
25  // B team sleeps on the wakeup_b_team condition variable. The A
26  // team does work. Threads transition to the B team if they wake
27  // up and find that a_team_size > target_a_team_size. Threads
28  // move into the A team whenever they wake up and find that
29  // a_team_size < target_a_team_size.
30  int a_team_size, target_a_team_size;
31 
32  // Broadcast when a job completes.
33  halide_cond wakeup_owners;
34 
35  // Broadcast whenever items are added to the work queue.
36  halide_cond wakeup_a_team;

39  // more threads are required than are currently in the A team.

91 WEAK void worker_thread_already_locked(work *owned_job) {
92  // If I'm a job owner, then I was the thread that called
93  // do_par_for, and I should only stay in this function until my
94  // job is complete. If I'm a lowly worker thread, I should stay in
95  // this function as long as the work queue is running.
96  while (owned_job != NULL ? owned_job->running()
97  : work_queue.running()) {
98 
99  if (work_queue.jobs == NULL) {
100  if (owned_job) {
101  // There are no jobs pending. Wait for the last worker
102  // to signal that the job is finished.
103  halide_cond_wait(&work_queue.wakeup_owners, &work_queue.mutex);
104  } else if (work_queue.a_team_size <= work_queue.target_a_team_size) {
105  // There are no jobs pending. Wait until more jobs are enqueued.
106  halide_cond_wait(&work_queue.wakeup_a_team, &work_queue.mutex);
107  } else {
108  // There are no jobs pending, and there are too many
109  // threads in the A team. Transition to the B team
110  // until the wakeup_b_team condition is fired.
111  work_queue.a_team_size--;
112  halide_cond_wait(&work_queue.wakeup_b_team, &work_queue.mutex);
113  work_queue.a_team_size++;
114  }
115  } else {
116  // Grab the next job.
117  work *job = work_queue.jobs;
118 
119  // Claim a task from it.
120  work myjob = *job;
121  job->next++;
122 
123  // If there were no more tasks pending for this job,
124  // remove it from the stack.
125  if (job->next == job->max) {
126  work_queue.jobs = job->next_job;
127  }
128 
129  // Increment the active_worker count so that other threads
130  // are aware that this job is still in progress even
131  // though there are no outstanding tasks for it.
132  job->active_workers++;
133 
134  // Release the lock and do the task.
135  halide_mutex_unlock(&work_queue.mutex);
136  int result = halide_do_task(myjob.user_context, myjob.f, myjob.next,
137  myjob.closure);
138  halide_mutex_lock(&work_queue.mutex);
139 
140  // If this task failed, set the exit status on the job.
141  if (result) {
142  job->exit_status = result;
143  }
144 
145  // We are no longer active on this job
146  job->active_workers--;
147 
148  // If the job is done and I'm not the owner of it, wake up
149  // the owner.
150  if (!job->running() && job != owned_job) {
151  halide_cond_broadcast(&work_queue.wakeup_owners);
152  }
153  }
154  }
155 }

164 WEAK int default_do_par_for(void *user_context, halide_task_t f,
165  int min, int size, uint8_t *closure) {
166  // Grab the lock. If it hasn't been initialized yet, then the
167  // field will be zero-initialized because it's a static global.
168  halide_mutex_lock(&work_queue.mutex);
169 
170  if (!work_queue.initialized) {
171  work_queue.shutdown = false;
172  halide_cond_init(&work_queue.wakeup_owners);
173  halide_cond_init(&work_queue.wakeup_a_team);
174  halide_cond_init(&work_queue.wakeup_b_team);
175  work_queue.jobs = NULL;
176 
177  // Compute the desired number of threads to use. Other code
178  // can also mess with this value, but only when the work queue
179  // is locked.
180  if (!work_queue.desired_num_threads) {
181  work_queue.desired_num_threads = default_desired_num_threads();
182  }
183  work_queue.desired_num_threads = clamp_num_threads(work_queue.desired_num_threads);
184  work_queue.threads_created = 0;
185 
186  // Everyone starts on the a team.
187  work_queue.a_team_size = work_queue.desired_num_threads;
188 
189  work_queue.initialized = true;
190  }
191 
192  while (work_queue.threads_created < work_queue.desired_num_threads - 1) {
193  // We might need to make some new threads, if work_queue.desired_num_threads has
194  // increased.
195  work_queue.threads[work_queue.threads_created++] =
196  halide_spawn_thread(worker_thread, NULL);
197  }
198 
199  // Make the job.
200  work job;
201  job.f = f; // The job should call this function. It takes an index and a closure.
202  job.user_context = user_context;
203  job.next = min; // Start at this index.
204  job.max = min + size; // Keep going until one less than this index.
205  job.closure = closure; // Use this closure.
206  job.exit_status = 0; // The job hasn't failed yet
207  job.active_workers = 0; // Nobody is working on this yet
208 
209  if (!work_queue.jobs && size < work_queue.desired_num_threads) {
210  // If there's no nested parallelism happening and there are
211  // fewer tasks to do than threads, then set the target A team
212  // size so that some threads will put themselves to sleep
213  // until a larger job arrives.
214  work_queue.target_a_team_size = size;
215  } else {
216  // Otherwise the target A team size is
217  // desired_num_threads. This may still be less than
218  // threads_created if desired_num_threads has been reduced by
219  // other code.
220  work_queue.target_a_team_size = work_queue.desired_num_threads;
221  }
222 
223  // Push the job onto the stack.
224  job.next_job = work_queue.jobs;
225  work_queue.jobs = &job;
226 
227  // Wake up our A team.
228  halide_cond_broadcast(&work_queue.wakeup_a_team);
229 
230  // If there are fewer threads than we would like on the a team,
231  // wake up the b team too.
232  if (work_queue.target_a_team_size > work_queue.a_team_size) {
233  halide_cond_broadcast(&work_queue.wakeup_b_team);
234  }
235 
236  // Do some work myself.
237  worker_thread_already_locked(&job);
238 
239  halide_mutex_unlock(&work_queue.mutex);
240 
241  // Return zero if the job succeeded, otherwise return the exit
242  // status of one of the failing jobs (whichever one failed last).
243  return job.exit_status;
244 }

252 WEAK int halide_set_num_threads(int n) {
253  if (n < 0) {
254  halide_error(NULL, "halide_set_num_threads: must be >= 0.");
255  }
256  // Don't make this an atomic swap - we don't want to be changing
257  // the desired number of threads while another thread is in the
258  // middle of a sequence of non-atomic operations.
259  halide_mutex_lock(&work_queue.mutex);
260  if (n == 0) {
261  n = default_desired_num_threads();
262  }
263  int old = work_queue.desired_num_threads;
264  work_queue.desired_num_threads = clamp_num_threads(n);
265  halide_mutex_unlock(&work_queue.mutex);
266  return old;
267 }

269 WEAK void halide_shutdown_thread_pool() {
270  if (!work_queue.initialized) return;
271 
272  // Wake everyone up and tell them the party's over and it's time
273  // to go home
274  halide_mutex_lock(&work_queue.mutex);
275  work_queue.shutdown = true;
276  halide_cond_broadcast(&work_queue.wakeup_owners);
277  halide_cond_broadcast(&work_queue.wakeup_a_team);
278  halide_cond_broadcast(&work_queue.wakeup_b_team);
279  halide_mutex_unlock(&work_queue.mutex);
280 
281  // Wait until they leave
282  for (int i = 0; i < work_queue.threads_created; i++) {
283  halide_join_thread(work_queue.threads[i]);
284  }
285 
286  // Tidy up
287  halide_mutex_destroy(&work_queue.mutex);
288  halide_cond_destroy(&work_queue.wakeup_owners);
289  halide_cond_destroy(&work_queue.wakeup_a_team);
290  halide_cond_destroy(&work_queue.wakeup_b_team);
291  work_queue.initialized = false;
292 }
