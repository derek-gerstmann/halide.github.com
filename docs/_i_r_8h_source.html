00027 /** The abstract base classes for a node in the Halide IR. */
00028 struct IRNode {
00029 
00030     /** We use the visitor pattern to traverse IR nodes throughout the
00031      * compiler, so we have a virtual accept method which accepts
00032      * visitors.
00033      */
00034     virtual void accept(IRVisitor *v) const = 0;
00035     IRNode() {}
00036     virtual ~IRNode() {}
00037 
00038     /** These classes are all managed with intrusive reference
00039        counting, so we also track a reference count. It's mutable
00040        so that we can do reference counting even through const
00041        references to IR nodes. */
00042     mutable RefCount ref_count;
00043 
00044     /** Each IR node subclass should return some unique pointer. We
00045      * can compare these pointers to do runtime type
00046      * identification. We don't compile with rtti because that
00047      * injects run-time type identification stuff everywhere (and
00048      * often breaks when linking external libraries compiled
00049      * without it), and we only want it for IR nodes. */
00050     virtual const IRNodeType *type_info() const = 0;
00051 };
