00028 struct IRNode {
00029 
00030     /** We use the visitor pattern to traverse IR nodes throughout the
00031      * compiler, so we have a virtual accept method which accepts
00032      * visitors.

00035     IRNode() {}
00036     virtual ~IRNode() {}
00037 
00038     /** These classes are all managed with intrusive reference
00039        counting, so we also track a reference count. It's mutable
