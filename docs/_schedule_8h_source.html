25 /** Different ways to handle a tail case in a split when the
26  * factor does not provably divide the extent. */
27 enum class TailStrategy {
28  /** Round up the extent to be a multiple of the split
29  * factor. Not legal for RVars, as it would change the meaning
30  * of the algorithm. Pros: generates the simplest, fastest
31  * code. Cons: if used on a stage that reads from the input or
32  * writes to the output, constrains the input or output size
33  * to be a multiple of the split factor. */
34  RoundUp,
35 
36  /** Guard the inner loop with an if statement that prevents
37  * evaluation beyond the original extent. Always legal. The if
38  * statement is treated like a boundary condition, and
39  * factored out into a loop epilogue if possible. Pros: no
40  * redundant re-evaluation; does not constrain input our
41  * output sizes. Cons: increases code size due to separate
42  * tail-case handling; vectorization will scalarize in the tail
43  * case to handle the if statement. */
44  GuardWithIf,
45 
46  /** Prevent evaluation beyond the original extent by shifting
47  * the tail case inwards, re-evaluating some points near the
48  * end. Only legal for pure variables in pure definitions. If
49  * the inner loop is very simple, the tail case is treated
50  * like a boundary condition and factored out into an
51  * epilogue.
52  *
53  * This is a good trade-off between several factors. Like
54  * RoundUp, it supports vectorization well, because the inner
55  * loop is always a fixed size with no data-dependent
56  * branching. It increases code size slightly for inner loops
57  * due to the epilogue handling, but not for outer loops
58  * (e.g. loops over tiles). If used on a stage that reads from
59  * an input or writes to an output, this stategy only requires
60  * that the input/output extent be at least the split factor,
61  * instead of a multiple of the split factor as with RoundUp. */
62  ShiftInwards,
63 
64  /** For pure definitions use ShiftInwards. For pure vars in
65  * update definitions use RoundUp. For RVars in update
66  * definitions use GuardWithIf. */
67  Auto
68 };

107 /** A reference to a site in a Halide statement at the top of the
108  * body of a particular for loop. Evaluating a region of a halide
109  * function is done by generating a loop nest that spans its
110  * dimensions. We schedule the inputs to that function by
111  * recursively injecting realizations for them at particular sites
112  * in this loop nest. A LoopLevel identifies such a site. The site
113  * can either be a loop nest within all stages of a function
114  * or it can refer to a loop nest within a particular function's
115  * stage (initial definition or updates).
116  *
117  * Note that a LoopLevel is essentially a pointer to an underlying value;
118  * all copies of a LoopLevel refer to the same site, so mutating one copy
119  * (via the set() method) will effectively mutate all copies:
120  \code
121  Func f;
122  Var x;
123  LoopLevel a(f, x);
124  // Both a and b refer to LoopLevel(f, x)
125  LoopLevel b = a;
126  // Now both a and b refer to LoopLevel::root()
127  a.set(LoopLevel::root());
128  \endcode
129  * This is quite useful when splitting Halide code into utility libraries, as it allows
130  * a library to schedule code according to a caller's specifications, even if the caller
131  * hasn't fully defined its pipeline yet:
132  \code
133  Func demosaic(Func input,
134  LoopLevel intermed_compute_at,
135  LoopLevel intermed_store_at,
136  LoopLevel output_compute_at) {
137  Func intermed = ...;
138  Func output = ...;
139  intermed.compute_at(intermed_compute_at).store_at(intermed_store_at);
140  output.compute_at(output_compute_at);
141  return output;
142  }
143 
144  void process() {
145  // Note that these LoopLevels are all undefined when we pass them to demosaic()
146  LoopLevel intermed_compute_at, intermed_store_at, output_compute_at;
147  Func input = ...;
148  Func demosaiced = demosaic(input, intermed_compute_at, intermed_store_at, output_compute_at);
149  Func output = ...;
150 
151  // We need to ensure all LoopLevels have a well-defined value prior to lowering:
152  intermed_compute_at.set(LoopLevel(output, y));
153  intermed_store_at.set(LoopLevel(output, y));
154  output_compute_at.set(LoopLevel(output, x));
155  }
156  \endcode
157  */
