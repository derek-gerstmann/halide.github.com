00015 /** A reference to a site in a Halide statement at the top of the
00016  * body of a particular for loop. Evaluating a region of a halide
00017  * function is done by generating a loop nest that spans its
00018  * dimensions. We schedule the inputs to that function by
00019  * recursively injecting realizations for them at particular sites
00020  * in this loop nest. A LoopLevel identifies such a site. */
00021 struct LoopLevel {
00022     std::string func, var;
00023 
00024     /** Identify the loop nest corresponding to some dimension of some function */
00025     LoopLevel(std::string f, std::string v) : func(f), var(v) {}
00026 
00027     /** Construct an empty LoopLevel, which is interpreted as
00028      * 'inline'. This is a special LoopLevel value that implies
00029      * that a function should be inlined away */
00030     LoopLevel() {}
00031 
00032     /** Test if a loop level corresponds to inlining the function */
00033     bool is_inline() const {return var.empty();}
00034 
00035     /** root is a special LoopLevel value which represents the
00036      * location outside of all for loops */
00037     static LoopLevel root() {
00038         return LoopLevel("", "__root");
00039     }
00040     /** Test if a loop level is 'root', which describes the site
00041      * outside of all for loops */
00042     bool is_root() const {return var == "__root";}
00043 
00044     /** Compare this loop level against the variable name of a for
00045      * loop, to see if this loop level refers to the site
00046      * immediately inside this loop. */
00047     bool match(const std::string &loop) const {
00048         return starts_with(loop, func + ".") && ends_with(loop, "." + var);
00049     }
00050 
00051     bool match(const LoopLevel &other) const {
00052         return (func == other.func &&
00053                 (var == other.var ||
00054                  ends_with(var, "." + other.var) ||
00055                  ends_with(other.var, "." + var)));
00056     }
00057 
00058     /** Check if two loop levels are exactly the same. */
00059     bool operator==(const LoopLevel &other) const {
00060         return func == other.func && var == other.var;
00061     }
