17/** \file
18 * Define a simple thread pool utility that is modeled on the api of
19 * std::async(); since implementation details of std::async
20 * can vary considerably, with no control over thread spawning, this class
21 * allows us to use the same model but with precise control over thread usage.
22 *
23 * A ThreadPool is created with a specific number of threads, which will never
24 * vary over the life of the ThreadPool. (If created without a specific number
25 * of threads, it will attempt to use threads == number-of-cores.)
26 *
27 * Each async request will go into a queue, and will be serviced by the next
28 * available thread from the pool.
29 *
30 * The ThreadPool's dtor will block until all currently-executing tasks
31 * to finish (but won't schedule any more).
32 *
33 * Note that this is a fairly simpleminded ThreadPool, meant for tasks
34 * that are fairly coarse (e.g. different tasks in a test); it is specifically
35 * *not* intended to be the underlying implementation for Halide runtime threads
36 */

53 std::queue<Job> jobs;
54
55 // Broadcast whenever items are added to the Job queue.
56 std::condition_variable wakeup_threads;
57
58 // Keep track of threads so they can be joined at shutdown
59 std::vector<std::thread> threads;

61 // True if the pool is shutting down.
62 bool shutting_down{false};

64 void worker_thread() {
65 std::unique_lock<std::mutex> unique_lock(mutex);
66 while (!shutting_down) {
67 if (jobs.empty()) {
68 // There are no jobs pending. Wait until more jobs are enqueued.
69 wakeup_threads.wait(unique_lock);
70 } else {
71 // Grab the next job.
72 Job cur_job = std::move(jobs.front());
73 jobs.pop();
74 cur_job.run_unlocked(unique_lock);
75 }
76 }
77 }

121 Job job;
