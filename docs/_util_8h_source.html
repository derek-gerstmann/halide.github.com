109 /** Perform a left fold of a vector. Returns a default-constructed
110  * vector element if the vector is empty. Similar to std::accumulate
111  * but with a less clunky syntax. */
112 template<typename T, typename Fn>
113 T fold_left(const std::vector<T> &vec, Fn f) {
114  T result;
115  if (vec.empty()) {
116  return result;
117  }
118  result = vec[0];
119  for (size_t i = 1; i < vec.size(); i++) {
120  result = f(result, vec[i]);
121  }
122  return result;
123 }

125 /** Returns a right fold of a vector. Returns a default-constructed
126  * vector element if the vector is empty. */
127 template<typename T, typename Fn>
128 T fold_right(const std::vector<T> &vec, Fn f) {
129  T result;
130  if (vec.empty()) {
131  return result;
132  }
133  result = vec.back();
134  for (size_t i = vec.size()-1; i > 0; i--) {
135  result = f(vec[i-1], result);
136  }
137  return result;
138 }
