00035     /** Prepare the buffer to be used as an image. Makes sure that the
00036      * cached strides are correct, and that the image data is on the
00037      * host. */
00038     void prepare_for_direct_pixel_access() {
00039         // Make sure buffer has been copied to host. This is a no-op
00040         // if there's no device involved.
00041         buffer.copy_to_host();
00042 
00043         // We're probably about to modify the pixels, so to be
00044         // conservative we'd better set host dirty. If you're sure
00045         // you're not going to modify this memory via the Image
00046         // object, then you can call set_host_dirty(false) on the
00047         // underlying buffer.
00048         buffer.set_host_dirty(true);
00049 
00050         if (buffer.defined()) {
00051             origin = (T *)buffer.host_ptr();
00052             stride_0 = buffer.stride(0);
00053             stride_1 = buffer.stride(1);
00054             stride_2 = buffer.stride(2);
00055             stride_3 = buffer.stride(3);
00056             // The host pointer points to the mins vec, but we want to
00057             // point to the origin of the coordinate system.
00058             origin -= (buffer.min(0) * stride_0 +
00059                        buffer.min(1) * stride_1 +
00060                        buffer.min(2) * stride_2 +
00061                        buffer.min(3) * stride_3);
00062             dims = buffer.dimensions();
00063         } else {
00064             origin = NULL;
00065             stride_0 = stride_1 = stride_2 = stride_3 = 0;
00066             dims = 0;
00067         }
00068     }
