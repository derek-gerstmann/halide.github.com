213/** A parallel task to be passed to halide_do_parallel_tasks. This
214 * task may recursively call halide_do_parallel_tasks, and there may
215 * be complex dependencies between seemingly unrelated tasks expressed
216 * using semaphores. If you are using a custom task system, care must

218 * respecting the static metadata at the end of the task struct.*/

221 // extent, a closure, and a task system pass through argument.

231 // function is called. Must be reacquired for every call made.

239 // A parallel task provides several pieces of metadata to prevent
240 // unbounded resource usage or deadlock.
241
242 // The first is the minimum number of execution contexts (call
243 // stacks or threads) necessary for the function to run to
244 // completion. This may be greater than one when there is nested
245 // parallelism with internal producer-consumer relationships
246 // (calling the function recursively spawns and blocks on parallel
247 // sub-tasks that communicate with each other via semaphores). If

255 // Halide pipeline, this may be an underestimate.

257
258 // The calls to the function should be in serial order from min to min+extent-1, with only
259 // one executing at a time. If false, any order is fine, and
260 // concurrency is fine.
261 bool serial;
262};

443/** A runtime tag for a type in the halide type system. Can be ints,
444 * unsigned ints, or floats of various bit-widths (the 'bits'

463 HALIDE_ATTRIBUTE_ALIGN(2)

467 /** Construct a runtime representation of a Halide type from:

469 * bits: The bit size of one element.
470 * lanes: The number of vector elements in the type. */
471 HALIDE_ALWAYS_INLINE constexpr halide_type_t(halide_type_code_t code, uint8_t bits, uint16_t lanes = 1)

486 return with_lanes(1);

488 /** Compare two types for equality. */
489 HALIDE_ALWAYS_INLINE constexpr bool operator==(const halide_type_t &other) const {
490 return as_u32() == other.as_u32();
491 }
492

494 return !(*this == other);
495 }

498 return as_u32() < other.as_u32();

501 /** Size in bytes for a single element, even if width is not 1, of this type. */
502 HALIDE_ALWAYS_INLINE constexpr int bytes() const {
503 return (bits + 7) / 8;
504 }

506 HALIDE_ALWAYS_INLINE constexpr uint32_t as_u32() const {
507 // Note that this produces a result that is identical to memcpy'ing 'this'
508 // into a u32 (on a little-endian machine, anyway), and at -O1 or greater
509 // on Clang, the compiler knows this and optimizes this into a single 32-bit move.
510 // (At -O0 it will look awful.)
511 return static_cast<uint8_t>(code) |
512 (static_cast<uint16_t>(bits) << 8) |
513 (static_cast<uint32_t>(lanes) << 16);
514 }
515#endif
516};

584 * sequence of trace packets. The header for a trace packet is defined
585 * below. If the trace is going to be large, you may want to make the
586 * file a named pipe, and then read from that pipe into gzip.
587 *
588 * halide_trace returns a unique ID which will be passed to future
589 * events that "belong" to the earlier event as the parent id. The
590 * ownership hierarchy looks like:
591 *
592 * begin_pipeline
593 * +--trace_tag (if any)
594 * +--trace_tag (if any)
595 * ...
596 * +--begin_realization
597 * | +--produce
598 * | | +--load/store
599 * | | +--end_produce
600 * | +--consume
601 * | | +--load
602 * | | +--end_consume
603 * | +--end_realization
604 * +--end_pipeline
605 *
606 * Threading means that ownership cannot be inferred from the ordering
607 * of events. There can be many active realizations of a given
608 * function, or many active productions for a single
609 * realization. Within a single production, the ordering of events is
610 * meaningful.
611 *
612 * Note that all trace_tag events (if any) will occur just after the begin_pipeline
613 * event, but before any begin_realization events. All trace_tags for a given Func
614 * will be emitted in the order added.
615 */
616// @}
617extern int32_t halide_trace(void *user_context, const struct halide_trace_event_t *event);

1522 * If all strides are positive, equal to zero.
1523 * Offset is in elements, not bytes.
1524 * Unlike begin(), this is ok to call on an unallocated buffer. */
