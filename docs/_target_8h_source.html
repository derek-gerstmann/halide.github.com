224  /** Given a data type, return an estimate of the "natural" vector size
225  * for that data type when compiling for this Target. */
226  int natural_vector_size(Halide::Type t) const {
227  user_assert(os != OSUnknown && arch != ArchUnknown && bits != 0)
228  << "natural_vector_size cannot be used on a Target with Unknown values.\n";
229 
230  const bool is_integer = t.is_int() || t.is_uint();
231  const int data_size = t.bytes();
232 
233  if (arch == Target::Hexagon) {
234  if (is_integer) {
235  // HVX is either 64 or 128 *byte* vector size.
236  if (has_feature(Halide::Target::HVX_128)) {
237  return 128 / data_size;
238  } else if (has_feature(Halide::Target::HVX_64)) {
239  return 64 / data_size;
240  } else {
241  user_error << "Target uses hexagon arch without hvx_128 or hvx_64 set.\n";
242  return 0;
243  }
244  } else {
245  // HVX does not have vector float instructions.
246  return 1;
247  }
248  } else if (arch == Target::X86) {
249  if (is_integer && (has_feature(Halide::Target::AVX512_Skylake) ||
250  has_feature(Halide::Target::AVX512_Cannonlake))) {
251  // AVX512BW exists on Skylake and Cannonlake
252  return 64 / data_size;
253  } else if (t.is_float() && (has_feature(Halide::Target::AVX512) ||
254  has_feature(Halide::Target::AVX512_KNL) ||
255  has_feature(Halide::Target::AVX512_Skylake) ||
256  has_feature(Halide::Target::AVX512_Cannonlake))) {
257  // AVX512F is on all AVX512 architectures
258  return 64 / data_size;
259  } else if (has_feature(Halide::Target::AVX2)) {
260  // AVX2 uses 256-bit vectors for everything.
261  return 32 / data_size;
262  } else if (!is_integer && has_feature(Halide::Target::AVX)) {
263  // AVX 1 has 256-bit vectors for float, but not for
264  // integer instructions.
265  return 32 / data_size;
266  } else {
267  // SSE was all 128-bit. We ignore MMX.
268  return 16 / data_size;
269  }
270  } else {
271  // Assume 128-bit vectors on other targets.
272  return 16 / data_size;
273  }
274  }
