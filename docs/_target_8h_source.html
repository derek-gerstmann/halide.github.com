167  /**
168  * Parse the contents of 'target' and merge into 'this',
169  * replacing only the parts that are specified. (e.g., if 'target' specifies
170  * only an arch, only the arch field of 'this' will be changed, leaving
171  * the other fields untouched). Any features specified in 'target'
172  * are added to 'this', whether or not originally present.
173  *
174  * If the string contains unknown tokens, or multiple tokens of the
175  * same category (e.g. multiple arch values), return false
176  * (possibly leaving 'this' munged). (Multiple feature specifications
177  * will not cause a failure.)
178  *
179  * If 'target' contains "host" as the first token, it replaces the entire
180  * contents of 'this' with get_host_target(), then proceeds to parse the
181  * remaining tokens (allowing for things like "host-opencl" to mean
182  * "host configuration, but with opencl added").
183  *
184  * Note that unlike parse_from_string(), this will never print to cerr or
185  * assert in the event of a parse failure. Note also that an empty target
186  * string is essentially a no-op, leaving 'this' unaffected.
187  */
188  EXPORT bool merge_string(const std::string &target);
189 
190  /**

198  /** Given a data type, return an estimate of the "natural" vector size
199  * for that data type when compiling for this Target. */
200  int natural_vector_size(Halide::Type t) const {
201  const bool is_avx2 = has_feature(Halide::Target::AVX2);
202  const bool is_avx = has_feature(Halide::Target::AVX) && !is_avx2;
203  const bool is_integer = t.is_int() || t.is_uint();
204 
205  // AVX has 256-bit SIMD registers, other existing targets have 128-bit ones.
206  // However, AVX has a very limited complement of integer instructions;
207  // restricting us to SSE4.1 size for integer operations produces much
208  // better performance. (AVX2 does have good integer operations for 256-bit
209  // registers.)
210  const int vector_byte_size = (is_avx2 || (is_avx && !is_integer)) ? 32 : 16;
211  const int data_size = t.bits / 8;
212  return vector_byte_size / data_size;
213  }
