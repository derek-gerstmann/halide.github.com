11 // A host <-> dev copy should be done with the fewest possible number
12 // of contiguous copies to minimize driver overhead. If our
13 // halide_buffer_t has strides larger than its extents (e.g. because
14 // it represents a sub-region of a larger halide_buffer_t) we can't
15 // safely copy it back and forth using a single contiguous copy,
16 // because we'd clobber in-between values that another thread might be
17 // using. In the best case we can do a single contiguous copy, but in
18 // the worst case we need to individually copy over every pixel.
19 //
20 // This problem is made extra difficult by the fact that the ordering
21 // of the dimensions in a halide_buffer_t doesn't relate to memory layout at
22 // all, so the strides could be in any order.
23 //
24 // We solve it by representing a copy job we need to perform as a
25 // device_copy struct. It describes a multi-dimensional array of
26 // copies to perform. Initially it describes copying over a single
27 // pixel at a time. We then try to discover contiguous groups of
28 // copies that can be coalesced into a single larger copy.
29 
30 // The struct that describes a host <-> dev copy to perform.
31 #define MAX_COPY_DIMS 16
32 struct device_copy {
33  uint64_t src, dst;
34  // The multidimensional array of contiguous copy tasks that need to be done.
35  uint64_t extent[MAX_COPY_DIMS];
36  // The strides (in bytes) that separate adjacent copy tasks in each dimension.
37  uint64_t stride_bytes[MAX_COPY_DIMS];
38  // How many contiguous bytes to copy per task
39  uint64_t chunk_size;
40 };

68 WEAK device_copy make_host_to_device_copy(const halide_buffer_t *buf) {
69  // Make a copy job representing copying the first pixel only.
70  device_copy c;
71  c.src = (uint64_t)buf->host;
72  c.dst = buf->device;
73  c.chunk_size = buf->type.bytes();
74  for (int i = 0; i < MAX_COPY_DIMS; i++) {
75  c.extent[i] = 1;
76  c.stride_bytes[i] = 0;
77  }
78

80  // This condition should also be checked for outside this fn.
81  device_copy zero = {0};
82  return zero;
83  }
84 
85  if (buf->type.bits == 0) {
86  // This buffer apparently represents no memory. Return a zero'd copy
87  // task.
88  device_copy zero = {0};
89  return zero;
90  }

92  // Now expand it to copy all the pixels (one at a time) by taking
93  // the extents and strides from the halide_buffer_t. Dimensions
94  // are added to the copy by inserting it such that the stride is
95  // in ascending order.
96  for (int i = 0; i < buf->dimensions; i++) {
97  // TODO: deal with negative strides.
98  uint64_t stride_bytes = buf->dim[i].stride * buf->type.bytes();
99  // Insert the dimension sorted into the buffer copy.
100  int insert;
101  for (insert = 0; insert < i; insert++) {
102  // If the stride is 0, we put it at the end because it can't be
103  // folded.
104  if (stride_bytes < c.stride_bytes[insert] && stride_bytes != 0) {
105  break;
106  }
107  }
108  for (int j = i; j > insert; j--) {
109  c.extent[j] = c.extent[j - 1];
110  c.stride_bytes[j] = c.stride_bytes[j - 1];
111  }
112  // If the stride is 0, only copy it once.
113  c.extent[insert] = stride_bytes != 0 ? buf->dim[i].extent : 1;
114  debug(NULL) << "c.extent[" << insert << "] = " << (int)(c.extent[insert]) << "\n";
115  c.stride_bytes[insert] = stride_bytes;
116  };
117 
118  // Attempt to fold contiguous dimensions into the chunk size. Since the
119  // dimensions are sorted by stride, and the strides must be greater than
120  // or equal to the chunk size, this means we can just delete the innermost
121  // dimension as long as its stride is equal to the chunk size.
122  while(c.chunk_size == c.stride_bytes[0]) {
123  // Fold the innermost dimension's extent into the chunk_size.
124  c.chunk_size *= c.extent[0];
125 
126  // Erase the innermost dimension from the list of dimensions to
127  // iterate over.
128  for (int j = 1; j < MAX_COPY_DIMS; j++) {
129  c.extent[j-1] = c.extent[j];
130  c.stride_bytes[j-1] = c.stride_bytes[j];
131  }
132  c.extent[MAX_COPY_DIMS-1] = 1;
133  c.stride_bytes[MAX_COPY_DIMS-1] = 0;
134  }
135  return c;
136 }
