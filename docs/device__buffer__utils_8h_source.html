32 // A host <-> dev copy should be done with the fewest possible number
33 // of contiguous copies to minimize driver overhead. If our buffer_t
34 // has strides larger than its extents (e.g. because it represents a
35 // sub-region of a larger buffer_t) we can't safely copy it back and
36 // forth using a single contiguous copy, because we'd clobber
37 // in-between values that another thread might be using. In the best
38 // case we can do a single contiguous copy, but in the worst case we
39 // need to individually copy over every pixel.
40 //
41 // This problem is made extra difficult by the fact that the ordering
42 // of the dimensions in a buffer_t doesn't relate to memory layout at
43 // all, so the strides could be in any order.
44 //
45 // We solve it by representing a copy job we need to perform as a
46 // device_copy struct. It describes a 4D array of copies to
47 // perform. Initially it describes copying over a single pixel at a
48 // time. We then try to discover contiguous groups of copies that can
49 // be coalesced into a single larger copy.
50 
51 // The struct that describes a host <-> dev copy to perform.
52 #define MAX_COPY_DIMS 4
53 struct device_copy {
54  uint64_t src, dst;
55  // The multidimensional array of contiguous copy tasks that need to be done.
56  uint64_t extent[MAX_COPY_DIMS];
57  // The strides (in bytes) that separate adjacent copy tasks in each dimension.
58  uint64_t stride_bytes[MAX_COPY_DIMS];
59  // How many contiguous bytes to copy per task
60  uint64_t chunk_size;
61 
62  void copy_memory(void *user_context) const;
63 };

90 WEAK device_copy make_host_to_device_copy(const buffer_t *buf) {
91  // Make a copy job representing copying the first pixel only.
92  device_copy c;
93  c.src = (uint64_t)buf->host;
94  c.dst = halide_get_device_handle(buf->dev);
95  c.chunk_size = buf->elem_size;
96  for (int i = 0; i < MAX_COPY_DIMS; i++) {
97  c.extent[i] = 1;
98  c.stride_bytes[i] = 0;
99  }
100 
101  if (buf->elem_size == 0) {
102  // This buffer apparently represents no memory. Return a zero'd copy
103  // task.
104  device_copy zero = {0};
105  return zero;
106  }
107 
108  // Now expand it to copy all the pixels (one at a time) by taking
109  // the extents and strides from the buffer_t. Dimensions are added
110  // to the copy by inserting it s.t. the stride is in ascending order.
111  for (int i = 0; i < 4 && buf->extent[i]; i++) {
112  // TODO: deal with negative strides.
113  uint64_t stride_bytes = buf->stride[i] * buf->elem_size;
114  // Insert the dimension sorted into the buffer copy.
115  int insert;
116  for (insert = 0; insert < i; insert++) {
117  // If the stride is 0, we put it at the end because it can't be
118  // folded.
119  if (stride_bytes < c.stride_bytes[insert] && stride_bytes != 0) {
120  break;
121  }
122  }
123  for (int j = i; j > insert; j--) {
124  c.extent[j] = c.extent[j - 1];
125  c.stride_bytes[j] = c.stride_bytes[j - 1];
126  }
127  // If the stride is 0, only copy it once.
128  c.extent[insert] = stride_bytes != 0 ? buf->extent[i] : 1;
129  c.stride_bytes[insert] = stride_bytes;
130  };
131 
132  // Attempt to fold contiguous dimensions into the chunk size. Since the
133  // dimensions are sorted by stride, and the strides must be greater than
134  // or equal to the chunk size, this means we can just delete the innermost
135  // dimension as long as its stride is equal to the chunk size.
136  while(c.chunk_size == c.stride_bytes[0]) {
137  // Fold the innermost dimension's extent into the chunk_size.
138  c.chunk_size *= c.extent[0];
139 
140  // Erase the innermost dimension from the list of dimensions to
141  // iterate over.
142  for (int j = 1; j < MAX_COPY_DIMS; j++) {
143  c.extent[j-1] = c.extent[j];
144  c.stride_bytes[j-1] = c.stride_bytes[j];
145  }
146  c.extent[MAX_COPY_DIMS-1] = 1;
147  c.stride_bytes[MAX_COPY_DIMS-1] = 0;
148  }
149  return c;
