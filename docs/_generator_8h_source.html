4 /** \file
5  *
6  * Generator is a class used to encapsulate the building of Funcs in user
7  * pipelines. A Generator is agnostic to JIT vs AOT compilation; it can be used for
8  * either purpose, but is especially convenient to use for AOT compilation.
9  *
10  * A Generator explicitly declares the Inputs and Outputs associated for a given
11  * pipeline, and separates the code for constructing the outputs from the code from
12  * scheduling them. For instance:
13  *
14  * \code
15  * class Blur : public Generator<Blur> {
16  * public:
17  * Input<Func> input{"input", UInt(16), 2};
18  * Output<Func> output{"output", UInt(16), 2};
19  * void generate() {
20  * blur_x(x, y) = (input(x, y) + input(x+1, y) + input(x+2, y))/3;
21  * blur_y(x, y) = (blur_x(x, y) + blur_x(x, y+1) + blur_x(x, y+2))/3;
22  * output(x, y) = blur(x, y);
23  * }
24  * void schedule() {
25  * blur_y.split(y, y, yi, 8).parallel(y).vectorize(x, 8);
26  * blur_x.store_at(blur_y, y).compute_at(blur_y, yi).vectorize(x, 8);
27  * }
28  * private:
29  * Var x, y, xi, yi;
30  * Func blur_x, blur_y;
31  * };
32  * \endcode
33  *
34  * Halide can compile a Generator into the correct pipeline by introspecting these
35  * values and constructing an appropriate signature based on them.
36  *
37  * A Generator must provide implementations of two methods:
38  *
39  * - generate(), which must fill in all Output Func(s), but should not do any
40  * scheduling
41  * - schedule(), which should do scheduling for any intermediate and
42  * output Funcs
43  *
44  * Inputs can be any C++ scalar type:
45  *
46  * \code
47  * Input<float> radius{"radius"};
48  * Input<int32_t> increment{"increment"};
49  * \endcode
50  *
51  * An Input<Func> is (essentially) like an ImageParam, except that it may (or may
52  * not) not be backed by an actual buffer, and thus has no defined extents.
53  *
54  * \code
55  * Input<Func> input{"input", Float(32), 2};
56  * \endcode
57  *
58  * You can optionally make the type and/or dimensions of Input<Func> unspecified,
59  * in which case the value is simply inferred from the actual Funcs passed to them.
60  * Of course, if you specify an explicit Type or Dimension, we still require the
61  * input Func to match, or a compilation error results.
62  *
63  * \code
64  * Input<Func> input{ "input", 3 }; // require 3-dimensional Func,
65  * // but leave Type unspecified
66  * \endcode
67  *
68  * A Generator must explicitly list the output(s) it produces:
69  *
70  * \code
71  * Output<Func> output{"output", Float(32), 2};
72  * \endcode
73  *
74  * You can specify an output that returns a Tuple by specifying a list of Types:
75  *
76  * \code
77  * class Tupler : Generator<Tupler> {
78  * Input<Func> input{"input", Int(32), 2};
79  * Output<Func> output{"output", {Float(32), UInt(8)}, 2};
80  * void generate() {
81  * Var x, y;
82  * Expr a = cast<float>(input(x, y));
83  * Expr b = cast<uint8_t>(input(x, y));
84  * output(x, y) = Tuple(a, b);
85  * }
86  * };
87  * \endcode
88  *
89  * You can also specify Output<X> for any scalar type (except for Handle types);
90  * this is merely syntactic sugar on top of a zero-dimensional Func, but can be
91  * quite handy, especially when used with multiple outputs:
92  *
93  * \code
94  * Output<float> sum{"sum"}; // equivalent to Output<Func> {"sum", Float(32), 0}
95  * \endcode
96  *
97  * As with Input<Func>, you can optionally make the type and/or dimensions of an
98  * Output<Func> unspecified; any unspecified types must be resolved via an
99  * implicit GeneratorParam in order to use top-level compilation.
100  *
101  * You can also declare an *array* of Input or Output, by using an array type
102  * as the type parameter:
103  *
104  * \code
105  * // Takes exactly 3 images and outputs exactly 3 sums.
106  * class SumRowsAndColumns : Generator<SumRowsAndColumns> {
107  * Input<Func[3]> inputs{"inputs", Float(32), 2};
108  * Input<int32_t[2]> extents{"extents"};
109  * Output<Func[3]> sums{"sums", Float(32), 1};
110  * void generate() {
111  * assert(inputs.size() == sums.size());
112  * // assume all inputs are same extent
113  * Expr width = extent[0];
114  * Expr height = extent[1];
115  * for (size_t i = 0; i < inputs.size(); ++i) {
116  * RDom r(0, width, 0, height);
117  * sums[i]() = 0.f;
118  * sums[i]() += inputs[i](r.x, r.y);
119  * }
120  * }
121  * };
122  * \endcode
123  *
124  * You can also leave array size unspecified, in which case it will be inferred
125  * from the input vector, or (optionally) explicitly specified via a resize()
126  * method:
127  *
128  * \code
129  * class Pyramid : public Generator<Pyramid> {
130  * public:
131  * GeneratorParam<int32_t> levels{"levels", 10};
132  * Input<Func> input{ "input", Float(32), 2 };
133  * Output<Func[]> pyramid{ "pyramid", Float(32), 2 };
134  * void generate() {
135  * pyramid.resize(levels);
136  * pyramid[0](x, y) = input(x, y);
137  * for (int i = 1; i < pyramid.size(); i++) {
138  * pyramid[i](x, y) = (pyramid[i-1](2*x, 2*y) +
139  * pyramid[i-1](2*x+1, 2*y) +
140  * pyramid[i-1](2*x, 2*y+1) +
141  * pyramid[i-1](2*x+1, 2*y+1))/4;
142  * }
143  * }
144  * };
145  * \endcode
146  *
147  * A Generator can also be customized via compile-time parameters (GeneratorParams
148  * or ScheduleParams), which affect code generation. While a GeneratorParam can
149  * be used from anywhere inside a Generator (either the generate() or
150  * schedule() method), ScheduleParam should be accessed only within the
151  * schedule() method. (This is not currently a compile-time error but may become
152  * one in the future.)
153  *
154  * GeneratorParams, ScheduleParams, Inputs, and Outputs are (by convention) always
155  * public and always declared at the top of the Generator class, in the order
156  *
157  * \code
158  * GeneratorParam(s)
159  * ScheduleParam(s)
160  * Input<Func>(s)
161  * Input<non-Func>(s)
162  * Output<Func>(s)
163  * \endcode
164  *
165  * Note that the Inputs and Outputs will appear in the C function call in the order
166  * they are declared. All Input<Func> and Output<Func> are represented as buffer_t;
167  * all other Input<> are the appropriate C++ scalar type. (GeneratorParams are
168  * always referenced by name, not position, so their order is irrelevant.)
169  *
170  * All Inputs and Outputs must have explicit names, and all such names must match
171  * the regex [A-Za-z][A-Za-z_0-9]* (i.e., essentially a C/C++ variable name, with
172  * some extra restrictions on underscore use). By convention, the name should match
173  * the member-variable name.
174  *
175  * Generators are added to a global registry to simplify AOT build mechanics; this
176  * is done by simply using the HALIDE_REGISTER_GENERATOR macro at global scope:
177  *
178  * \code
179  * HALIDE_REGISTER_GENERATOR(ExampleGen, "jit_example")
180  * \endcode
181  *
182  * The registered name of the Generator is provided must match the same rules as
183  * Input names, above.
184  *
185  * Note that a Generator is always executed with a specific Target assigned to it,
186  * that you can access via the get_target() method. (You should *not* use the
187  * global get_target_from_environment(), etc. methods provided in Target.h)
188  *
189  * (Note that there are older variations of Generator that differ from what's
190  * documented above; these are still supported but not described here. See
191  * https://github.com/halide/Halide/wiki/Old-Generator-Documentation for
192  * more information.)
193  */

223  * on any Parameter-based argument (i.e., Input<Buffer> and Output<Buffer>) to
224  * ensure that incompatible values aren't set.
225  *
226  * e.g.: if a Generator A requires stride[0] == 1,
227  * and Generator B uses Generator A via stub, but requires stride[0] == 4,
228  * we should be able to detect this at Halide compilation time, and fail immediately,
229  * rather than producing code that fails at runtime and/or runs slowly due to
230  * vectorization being unavailable.
231  *

244  * layers of Stubs in a single fused result.
