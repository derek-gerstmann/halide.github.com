4 /** \file
5  *
6  * Generator is a class used to encapsulate the building of Funcs in user pipelines.
7  * A Generator is agnostic to JIT vs AOT compilation; it can be used for either
8  * purpose, but is especially convenient to use for AOT compilation.
9  *
10  * A Generator automatically detects the run-time parameters (Param/ImageParams)
11  * associated with the Func and (for AOT code) produces a function signature
12  * with the correct params in the correct order.
13  *
14  * A Generator can also be customized via compile-time parameters (GeneratorParams),
15  * which affect code generation.
16  *
17  * GeneratorParams, ImageParams, and Params are (by convention)
18  * always public and always declared at the top of the Generator class,
19  * in the order
20  *
21  * GeneratorParam(s)
22  * ImageParam(s)
23  * Param(s)
24  *
25  * Preferred style is to use C++11 in-class initialization style, e.g.
26  * \code
27  * GeneratorParam<int> magic{"magic", 42};
28  * \endcode
29  *
30  * Note that the ImageParams/Params will appear in the C function
31  * call in the order they are declared. (GeneratorParams are always
32  * referenced by name, not position, so their order is irrelevant.)
33  *
34  * All Param variants declared as Generator members must have explicit
35  * names, and all such names must match the regex [A-Za-z][A-Za-z_0-9]*
36  * (i.e., essentially a C/C++ variable name, with some extra restrictions
37  * on underscore use). By convention, the name should match the member-variable name.
38  *
39  * Generators are usually added to a global registry to simplify AOT build mechanics;
40  * this is done by simply defining an instance of RegisterGenerator at static
41  * scope:
42  * \code
43  * RegisterGenerator<ExampleGen> register_jit_example{"jit_example"};
44  * \endcode
45  *
46  * The registered name of the Generator is provided as an argument
47  * (which must match the same rules as Param names, above).
48  *
49  * (If you are jitting, you may not need to bother registering your Generator,
50  * but it's considered best practice to always do so anyway.)
51  *
52  * Most Generator classes will only need to override the build() method,
53  * and perhaps declare a Param and/or GeneratorParam:
54  *
55  * \code
56  * class XorImage : public Generator<XorImage> {
57  * public:
58  * GeneratorParam<int> channels{"channels", 3};
59  * ImageParam input{UInt(8), 3, "input"};
60  * Param<uint8_t> mask{"mask"};
61  *
62  * Func build() override {
63  * Var x, y, c;
64  * Func f;
65  * f(x, y, c) = input(x, y, c) ^ mask;
66  * f.bound(c, 0, bound).reorder(c, x, y).unroll(c);
67  * return f;
68  * }
69  * };
70  * RegisterGenerator<XorImage> reg_xor{"xor_image"};
71  * \endcode
72  *
73  * By default, this code schedules itself for 3-channel (RGB) images;
74  * by changing the value of the "channels" GeneratorParam before calling
75  * build() we can produce code suited for different channel counts.
76  *
77  * Note that a Generator is always executed with a specific Target
78  * assigned to it, that you can access via the get_target() method.
79  * (You should *not* use the global get_target_from_environment(), etc.
80  * methods provided in Target.h)
81  */

121 /** GeneratorParam is a templated class that can be used to modify the behavior
122  * of the Generator at code-generation time. GeneratorParams are commonly
123  * specified in build files (e.g. Makefile) to customize the behavior of
124  * a given Generator, thus they have a very constrained set of types to allow
125  * for efficient specification via command-line flags. A GeneratorParm can be:
126  * - any float or int type.
127  * - bool
128  * - enum
129  * - Halide::Target
130  * - Halide::Type
131  * All GeneratorParams have a default value. Arithmetic types can also
132  * optionally specify min and max. Enum types must specify a string-to-value
133  * map.
134  *
135  * Halide::Type is treated as though it were an enum, with the mappings:
136  *
137  * "int8" Halide::Int(8)
138  * "int16" Halide::Int(16)
139  * "int32" Halide::Int(32)
140  * "uint8" Halide::UInt(8)
141  * "uint16" Halide::UInt(16)
142  * "uint32" Halide::UInt(32)
143  * "float32" Halide::Float(32)
144  * "float64" Halide::Float(64)
145  *
146  * No vector Types are currently supported by this mapping.
147  *
148  */
