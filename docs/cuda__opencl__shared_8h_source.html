00026 // A host <-> dev copy should be done with the fewest possible number
00027 // of contiguous copies to minimize driver overhead. If our buffer_t
00028 // has strides larger than its extents (e.g. because it represents a
00029 // sub-region of a larger buffer_t) we can't safely copy it back and
00030 // forth using a single contiguous copy, because we'd clobber
00031 // in-between values that another thread might be using.  In the best
00032 // case we can do a single contiguous copy, but in the worst case we
00033 // need to individually copy over every pixel.
00034 //
00035 // This problem is made extra difficult by the fact that the ordering
00036 // of the dimensions in a buffer_t doesn't relate to memory layout at
00037 // all, so the strides could be in any order.
00038 //
00039 // We solve it by representing a copy job we need to perform as a
00040 // dev_copy struct. It describes a 4D array of copies to
00041 // perform. Initially it describes copying over a single pixel at a
00042 // time. We then try to discover contiguous groups of copies that can
00043 // be coalesced into a single larger copy.
00044 
00045 // The struct that describes a host <-> dev copy to perform.
00046 #define MAX_COPY_DIMS 4
00047 struct dev_copy {
00048     uint64_t src, dst;
00049     // The multidimensional array of contiguous copy tasks that need to be done.
00050     uint64_t extent[MAX_COPY_DIMS];
00051     // The strides (in bytes) that separate adjacent copy tasks in each dimension.
00052     uint64_t stride_bytes[MAX_COPY_DIMS];
00053     // How many contiguous bytes to copy per task
00054     uint64_t chunk_size;
00055 };

00059     dev_copy c;

00070         // task.

00086                 break;

00101     // dimension as long as its stride is equal to the chunk size.

00104         c.chunk_size *= c.extent[0];

00107         // iterate over.

00110             c.stride_bytes[j-1] = c.stride_bytes[j];
00111         }
00112         c.extent[MAX_COPY_DIMS-1] = 1;
00113         c.stride_bytes[MAX_COPY_DIMS-1] = 0;
00114     }
00115     return c;
