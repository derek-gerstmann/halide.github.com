97  /** Calling rfactor() on an associative update definition a Func will split
98  * the update into an intermediate which computes the partial results and
99  * replaces the current update definition with a new definition which merges
100  * the partial results. If called on a init/pure definition, this will
101  * throw an error. rfactor() will automatically infer the associative reduction
102  * operator and identity of the operator. If it can't prove the operation
103  * is associative or if it cannot find an identity for that operator, this
104  * will throw an error. In addition, commutativity of the operator is required
105  * if rfactor() is called on the inner dimension but excluding the outer
106  * dimensions.
107  *
108  * rfactor() takes as input 'preserved', which is a list of <RVar, Var> pairs.
109  * The rvars not listed in 'preserved' are removed from the original Func and
110  * are lifted to the intermediate Func. The remaining rvars (the ones in
111  * 'preserved') are made pure in the intermediate Func. The intermediate Func's
112  * update definition inherits all scheduling directives (e.g. split,fuse, etc.)
113  * applied to the original Func's update definition. The loop order of the
114  * intermediate Func's update definition is the same as the original, although
115  * the RVars in 'preserved' are replaced by the new pure Vars. The loop order of the
116  * intermediate Func's init definition from innermost to outermost is the args'
117  * order of the original Func's init definition followed by the new pure Vars.
118  *
119  * The intermediate Func also inherits storage order from the original Func
120  * with the new pure Vars added to the outermost.
121  *
122  * For example, f.update(0).rfactor({{r.y, u}}) would rewrite a pipeline like this:
123  \code
124  f(x, y) = 0;
125  f(x, y) += g(r.x, r.y);
126  \endcode
127  * into a pipeline like this:
128  \code
129  f_intm(x, y, u) = 0;
130  f_intm(x, y, u) += g(r.x, u);
131 
132  f(x, y) = 0;
133  f(x, y) += f_intm(x, y, r.y);
134  \endcode
135  *
136  * This has a variety of uses. You can use it to split computation of an associative reduction:
137  \code
138  f(x, y) = 10;
139  RDom r(0, 96);
140  f(x, y) = max(f(x, y), g(x, y, r.x));
141  f.update(0).split(r.x, rxo, rxi, 8).reorder(y, x).parallel(x);
142  f.update(0).rfactor({{rxo, u}}).compute_root().parallel(u).update(0).parallel(u);
143  \endcode
144  *
145  *, which is equivalent to:
146  \code
147  parallel for u = 0 to 11:
148  for y:
149  for x:
150  f_intm(x, y, u) = -inf
151  parallel for x:
152  for y:
153  parallel for u = 0 to 11:
154  for rxi = 0 to 7:
155  f_intm(x, y, u) = max(f_intm(x, y, u), g(8*u + rxi))
156  for y:
157  for x:
158  f(x, y) = 10
159  parallel for x:
160  for y:
161  for rxo = 0 to 11:
162  f(x, y) = max(f(x, y), f_intm(x, y, u))
163  \endcode
164  *

1246  /** Specialize a Func. This creates a special-case version of the
1247  * Func where the given condition is true. The most effective
1248  * conditions are those of the form param == value, and boolean
1249  * Params. Consider a simple example:
1250  \code
1251  f(x) = x + select(cond, 0, 1);
1252  f.compute_root();
1253  \endcode
1254  * This is equivalent to:
1255  \code
1256  for (int x = 0; x < width; x++) {
1257  f[x] = x + (cond ? 0 : 1);
1258  }
1259  \endcode
1260  * Adding the scheduling directive:
1261  \code
1262  f.specialize(cond)
1263  \endcode
1264  * makes it equivalent to:
1265  \code
1266  if (cond) {
1267  for (int x = 0; x < width; x++) {
1268  f[x] = x;
1269  }
1270  } else {
1271  for (int x = 0; x < width; x++) {
1272  f[x] = x + 1;
1273  }
1274  }
1275  \endcode
1276  * Note that the inner loops have been simplified. In the first
1277  * path Halide knows that cond is true, and in the second path
1278  * Halide knows that it is false.
1279  *
1280  * The specialized version gets its own schedule, which inherits
1281  * every directive made about the parent Func's schedule so far
1282  * except for its specializations. This method returns a handle to
1283  * the new schedule. If you wish to retrieve the specialized
1284  * sub-schedule again later, you can call this method with the
1285  * same condition. Consider the following example of scheduling
1286  * the specialized version:
1287  *
1288  \code
1289  f(x) = x;
1290  f.compute_root();
1291  f.specialize(width > 1).unroll(x, 2);
1292  \endcode
1293  * Assuming for simplicity that width is even, this is equivalent to:
1294  \code
1295  if (width > 1) {
1296  for (int x = 0; x < width/2; x++) {
1297  f[2*x] = 2*x;
1298  f[2*x + 1] = 2*x + 1;
1299  }
1300  } else {
1301  for (int x = 0; x < width/2; x++) {
1302  f[x] = x;
1303  }
1304  }
1305  \endcode
1306  * For this case, it may be better to schedule the un-specialized
1307  * case instead:
1308  \code
1309  f(x) = x;
1310  f.compute_root();
1311  f.specialize(width == 1); // Creates a copy of the schedule so far.
1312  f.unroll(x, 2); // Only applies to the unspecialized case.
1313  \endcode
1314  * This is equivalent to:
1315  \code
1316  if (width == 1) {
1317  f[0] = 0;
1318  } else {
1319  for (int x = 0; x < width/2; x++) {
1320  f[2*x] = 2*x;
1321  f[2*x + 1] = 2*x + 1;
1322  }
1323  }
1324  \endcode
1325  * This can be a good way to write a pipeline that splits,
1326  * vectorizes, or tiles, but can still handle small inputs.
1327  *
1328  * If a Func has several specializations, the first matching one
1329  * will be used, so the order in which you define specializations
1330  * is significant. For example:
1331  *
1332  \code
1333  f(x) = x + select(cond1, a, b) - select(cond2, c, d);
1334  f.specialize(cond1);
1335  f.specialize(cond2);
1336  \endcode
1337  * is equivalent to:
1338  \code
1339  if (cond1) {
1340  for (int x = 0; x < width; x++) {
1341  f[x] = x + a - (cond2 ? c : d);
1342  }
1343  } else if (cond2) {
1344  for (int x = 0; x < width; x++) {
1345  f[x] = x + b - c;
1346  }
1347  } else {
1348  for (int x = 0; x < width; x++) {
1349  f[x] = x + b - d;
1350  }
1351  }
1352  \endcode
1353  *
1354  * Specializations may in turn be specialized, which creates a
1355  * nested if statement in the generated code.
1356  *
1357  \code
1358  f(x) = x + select(cond1, a, b) - select(cond2, c, d);
1359  f.specialize(cond1).specialize(cond2);
1360  \endcode
1361  * This is equivalent to:
1362  \code
1363  if (cond1) {
1364  if (cond2) {
1365  for (int x = 0; x < width; x++) {
1366  f[x] = x + a - c;
1367  }
1368  } else {
1369  for (int x = 0; x < width; x++) {
1370  f[x] = x + a - d;
1371  }
1372  }
1373  } else {
1374  for (int x = 0; x < width; x++) {
1375  f[x] = x + b - (cond2 ? c : d);
1376  }
1377  }
1378  \endcode
1379  * To create a 4-way if statement that simplifies away all of the
1380  * ternary operators above, you could say:
1381  \code
1382  f.specialize(cond1).specialize(cond2);
1383  f.specialize(cond2);
1384  \endcode
1385  * or
1386  \code
1387  f.specialize(cond1 && cond2);
1388  f.specialize(cond1);
1389  f.specialize(cond2);
1390  \endcode
1391  *
1392  * Any prior Func which is compute_at some variable of this Func
1393  * gets separately included in all paths of the generated if
1394  * statement. The Var in the compute_at call to must exist in all
1395  * paths, but it may have been generated via a different path of
1396  * splits, fuses, and renames. This can be used somewhat
1397  * creatively. Consider the following code:
1398  \code
1399  g(x, y) = 8*x;
1400  f(x, y) = g(x, y) + 1;
1401  f.compute_root().specialize(cond);
1402  Var g_loop;
1403  f.specialize(cond).rename(y, g_loop);
1404  f.rename(x, g_loop);
1405  g.compute_at(f, g_loop);
1406  \endcode
1407  * When cond is true, this is equivalent to g.compute_at(f,y).
1408  * When it is false, this is equivalent to g.compute_at(f,x).
