777  /** Make a new image which is a deep copy of this image. Use crop
778  * or slice followed by copy to make a copy of only a portion of
779  * the image. The new image uses the same memory layout as the
780  * original, with holes compacted away. */
781  Buffer<T, D> copy(void *(*allocate_fn)(size_t) = nullptr,
782  void (*deallocate_fn)(void *) = nullptr) const {
783  Buffer<T, D> src = *this;
784 
785  // Reorder the dimensions of src to have strides in increasing order
786  int swaps[(D*(D+1))/2];
787  int swaps_idx = 0;
788  for (int i = dimensions()-1; i > 0; i--) {
789  for (int j = i; j > 0; j--) {
790  if (src.dim(j-1).stride() > src.dim(j).stride()) {
791  src.transpose(j-1, j);
792  swaps[swaps_idx++] = j;
793  }
794  }
795  }
796 
797  // Make a copy of it using this dimension ordering
798  Buffer<T, D> dst = src;
799  dst.allocate(allocate_fn, deallocate_fn);
800 
801  // Concatenate dense inner dimensions into contiguous memcpy tasks
802  Buffer<T, D> src_slice = src;
803  Buffer<T, D> dst_slice = dst;
804  int64_t slice_size = 1;
805  while (src_slice.dimensions && src_slice.dim(0).stride() == slice_size) {
806  assert(dst_slice.dim(0).stride() == slice_size);
807  slice_size *= src_slice.dim(0).stride();
808  src_slice = src_slice.sliced(0, src_slice.dim(0).min());
809  dst_slice = dst_slice.sliced(0, dst_slice.dim(0).min());
810  }
811 
812  slice_size *= buf.elem_size;
813  // Do the memcpys
814  src_slice.for_each_element([&](const int *pos) {
815  memcpy(&dst_slice(pos), &src_slice(pos), slice_size);
816  });
817 
818  // Undo the dimension reordering
819  while (swaps_idx > 0) {
820  int j = swaps[--swaps_idx];
821  dst.transpose(j-1, j);
822  }
823 
824  return dst;
825  }
