62 /** A multi-dimensional domain over which to iterate. Used when
63  * defining functions with update definitions.
64  *
65  * An reduction is a function with a two-part definition. It has an
66  * initial value, which looks much like a pure function, and an update
67  * definition, which may refer to some RDom. Evaluating such a
68  * function first initializes it over the required domain (which is
69  * inferred based on usage), and then runs update rule for all points
70  * in the RDom. For example:
71  *
72  \code
73  Func f;
74  Var x;
75  RDom r(0, 10);
76  f(x) = x; // the initial value
77  f(r) = f(r) * 2;
78  Buffer<int> result = f.realize(10);
79  \endcode
80  *
81  * This function creates a single-dimensional buffer of size 10, in
82  * which element x contains the value x*2. Internally, first the
83  * initialization rule fills in x at every site, and then the update
84  * definition doubles every site.
85  *
86  * One use of reductions is to build a function recursively (pure
87  * functions in halide cannot be recursive). For example, this
88  * function fills in an array with the first 20 fibonacci numbers:
89  *
90  \code
91  Func f;
92  Var x;
93  RDom r(2, 18);
94  f(x) = 1;
95  f(r) = f(r-1) + f(r-2);
96  \endcode
97  *
98  * Another use of reductions is to perform scattering operations, as
99  * unlike a pure function declaration, the left-hand-side of an update
100  * definition may contain general expressions:
101  *
102  \code
103  ImageParam input(UInt(8), 2);
104  Func histogram;
105  Var x;
106  RDom r(input); // Iterate over all pixels in the input
107  histogram(x) = 0;
108  histogram(input(r.x, r.y)) = histogram(input(r.x, r.y)) + 1;
109  \endcode
110  *
111  * An update definition may also be multi-dimensional. This example
112  * computes a summed-area table by first summing horizontally and then
113  * vertically:
114  *
115  \code
116  ImageParam input(Float(32), 2);
117  Func sum_x, sum_y;
118  Var x, y;
119  RDom r(input);
120  sum_x(x, y) = input(x, y);
121  sum_x(r.x, r.y) = sum_x(r.x, r.y) + sum_x(r.x-1, r.y);
122  sum_y(x, y) = sum_x(x, y);
123  sum_y(r.x, r.y) = sum_y(r.x, r.y) + sum_y(r.x, r.y-1);
124  \endcode
125  *
126  * You can also mix pure dimensions with reduction variables. In the
127  * previous example, note that there's no need for the y coordinate in
128  * sum_x to be traversed serially. The sum within each row is entirely
129  * independent. The rows could be computed in parallel, or in a
130  * different order, without changing the meaning. Therefore, we can
131  * instead write this definition as follows:
132  *
133  \code
134  ImageParam input(Float(32), 2);
135  Func sum_x, sum_y;
136  Var x, y;
137  RDom r(input);
138  sum_x(x, y) = input(x, y);
139  sum_x(r.x, y) = sum_x(r.x, y) + sum_x(r.x-1, y);
140  sum_y(x, y) = sum_x(x, y);
141  sum_y(x, r.y) = sum_y(x, r.y) + sum_y(x, r.y-1);
142  \endcode
143  *
144  * This lets us schedule it more flexibly. You can now parallelize the
145  * update step of sum_x over y by calling:
146  \code
147  sum_x.update().parallel(y).
148  \endcode
149  *
150  * Note that calling sum_x.parallel(y) only parallelizes the
151  * initialization step, and not the update step! Scheduling the update
152  * step of a reduction must be done using the handle returned by
153  * \ref Func::update(). This code parallelizes both the initialization
154  * step and the update step:
155  *
156  \code
157  sum_x.parallel(y);
158  sum_x.update().parallel(y);
159  \endcode
160  *
161  * When you mix reduction variables and pure dimensions, the reduction
162  * domain is traversed outermost. That is, for each point in the
163  * reduction domain, the inferred pure domain is traversed in its
164  * entirety. For the above example, this means that sum_x walks down
165  * the columns, and sum_y walks along the rows. This may not be
166  * cache-coherent. You may try reordering these dimensions using the
167  * schedule, but Halide will return an error if it decides that this
168  * risks changing the meaning of your function. The solution lies in
169  * clever scheduling. If we say:
170  *
171  \code
172  sum_x.compute_at(sum_y, y);
173  \endcode
174  *
175  * Then the sum in x is computed only as necessary for each scanline
176  * of the sum in y. This not only results in sum_x walking along the
177  * rows, it also improves the locality of the entire pipeline.
178  */

246  /** Add a predicate to the RDom. An RDom may have multiple
247  * predicates associated with it. An update definition that uses
248  * an RDom only iterates over the subset points in the domain for
249  * which all of its predicates are true. The predicate expression
250  * obeys the same rules as the expressions used on the
251  * right-hand-side of the corresponding update definition. It may
252  * refer to the RDom's variables and free variables in the Func's
253  * update definition. It may include calls to other Funcs, or make
254  * recursive calls to the same Func. This permits iteration over
255  * non-rectangular domains, or domains with sizes that vary with
256  * some free variable, or domains with shapes determined by some
257  * other Func.
258  *
259  * Note that once RDom is used in the update definition of some
260  * Func, no new predicates can be added to the RDom.
261  *
262  * Consider a simple example:
263  \code
264  RDom r(0, 20, 0, 20);
265  r.where(r.x < r.y);
266  r.where(r.x == 10);
267  r.where(r.y > 13);
268  f(r.x, r.y) += 1;
269  \endcode
270  * This is equivalent to:
271  \code
272  for (int r.y = 0; r.y < 20; r.y++) {
273  if (r.y > 13) {
274  for (int r.x = 0; r.x < 20; r.x++) {
275  if (r.x == 10) {
276  if (r.x < r.y) {
277  f[r.x, r.y] += 1;
278  }
279  }
280  }
281  }
282  }
283  \endcode
284  *
285  * Where possible Halide restricts the range of the containing for
286  * loops to avoid the cases where the predicate is false so that
287  * the if statement can be removed entirely. The case above would
288  * be further simplified into:
289  *
290  \code
291  for (int r.y = 14; r.y < 20; r.y++) {
292  f[r.x, r.y] += 1;
293  }
294  \endcode
295  *
296  * In general, the predicates that we can simplify away by
297  * restricting loop ranges are inequalities that compare an inner
298  * Var or RVar to some expression in outer Vars or RVars.
299  *
300  * You can also pack multiple conditions into one predicate like so:
301  *
302  \code
303  RDom r(0, 20, 0, 20);
304  r.where((r.x < r.y) && (r.x == 10) && (r.y > 13));
305  f(r.x, r.y) += 1;
306  \endcode
307  *
308  */
309  EXPORT void where(Expr predicate);
