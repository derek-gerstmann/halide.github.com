197  // determine their span.
198  char *min_addr = NULL, *max_addr = NULL;
199  for (int i = 0; i < num_sections(); i++) {
200  section_header_t *sec = get_section(i);
201  if (is_section_writeable(sec)) {
202  char *start = get_section_start(sec);
203  char *end = start + get_section_size(sec);
204  if (min_addr == NULL || start < min_addr) {
205  min_addr = start;
206  }
207  if (max_addr == NULL || end > max_addr) {
208  max_addr = end;
209  }
210  }
211  }
212  size_t size_to_copy = (max_addr - min_addr);
213 
214  if (size_to_copy == 0) {
215  if (debug) log_printf("No writeable sections\n");
216  return;
217  }
218

220  // junk, e.g. if the first and last sections of the object
221  // file are writeable, but in practice we've found that the
222  // writeable sections are in one tight cluster. We don't want
223  // to copy over the sections individually, because some of
224  // them alias.

227  writeable_size = (size_to_copy + alignment - 1) & ~(alignment - 1);

232  if (!writeable_buf) {
233  fail(__LINE__);
234  return;
235  }
236 
237  if (debug) {
238  log_printf("Copying %d bytes of writeable data from %p to %p to a separate mapping of size %d at %p\n",

240  }

243  memcpy(writeable_buf, min_addr, size_to_copy);

254  if (failed) return;

437  // instruction encodings.

442  log_printf("\n");
443  }
444

447  // subinstructions. See section 3.7 in the
448  // programmer's reference.

452  if (debug) {
453  log_printf("Class: %x\n", iclass);
454  log_printf("Hi: ");
455  for (int i = 28; i >= 16; i--) {
456  log_printf("%d", (int)((inst >> i) & 1));
457  }
458  log_printf("\n");
459  log_printf("Lo: ");
460  for (int i = 12; i >= 0; i--) {
461  log_printf("%d", (int)((inst >> i) & 1));
462  }
463  log_printf("\n");
464  }
465 
466  // We only know how to do the ones where the high
467  // subinstruction is an immediate assignment. (marked
468  // as A in table 9-4 in the programmer's reference
469  // manual).
470  if (iclass < 3 || iclass > 7) {
471  fail(__LINE__);
472  return;
473  }
474 
475  // Pull out the subinstructions. They're the low 13
476  // bits of each half-word.
477  uint32_t hi = (inst >> 16) & ((1 << 13) - 1);
478  uint32_t lo = inst & ((1 << 13) - 1);
479 
480  // We only understand the ones where hi starts with 010
481  if ((hi >> 10) != 2) {
482  fail(__LINE__);
483  return;
484  }
485 
486  // Low 6 bits of val go in the following bits.
487  mask = 63 << 20;
488 
489  } else if ((inst >> 24) == 72) {

491  // 0100 1ii0 000i iiii PPit tttt iiii iiii
492  if (debug) log_printf("Instruction-specific case A\n");
493  mask = 0x061f20ff;
494  } else if ((inst >> 24) == 73) {

496  if (debug) log_printf("Instruction-specific case B\n");
497  mask = 0x061f3fe0;
498  } else if ((inst >> 24) == 120) {
499  // 0111 1000 ii-i iiii PPii iiii iiid dddd
500  if (debug) log_printf("Instruction-specific case C\n");
501  mask = 0x00df3fe0;
502  } else if ((inst >> 16) == 27209) {
503  // 0110 1010 0100 1001 PP-i iiii i--d dddd
504  mask = 0x00001f80;
505  } else if ((inst >> 25) == 72) {
506  // 1001 0ii0 101s ssss PPii iiii iiid dddd
507  // 1001 0ii1 000s ssss PPii iiii iiid dddd
508  mask = 0x06003fe0;
509  } else if ((inst >> 24) == 115 || (inst >> 24) == 124) {
510  // 0111 0011 -10sssss PP1iiiii iiiddddd
511  // 0111 0011 -11sssss PP1iiiii iiiddddd

516  // 0111 1100 0IIIIIII PPIiiiii iiiddddd
517  // 0111 0011 -11sssss PP1iiiii iiiddddd
518  mask = 0x00001fe0;
519 
520  } else {
521  log_printf("Unhandled!\n");
522  fail(__LINE__);
523  return;
524  }
525  }
526 
527  uintptr_t old_val = val;
528  bool consumed_every_bit = false;
529  for (int i = 0; i < 32; i++) {
530  if (mask & (1 << i)) {
531  if (inst & (1 << i)) {
532  // This bit should be zero in the unrelocated instruction
533  fail(__LINE__);
534  return;
535  }
536  // Consume a bit of val
537  int next_bit = val & 1;
538  if (is_signed) {
539  consumed_every_bit |= ((intptr_t)val) == -1;
540  val = ((intptr_t)val) >> 1;
541  } else {
542  val = ((uintptr_t)val) >> 1;
543  }
544  consumed_every_bit |= (val == 0);
545  inst |= (next_bit << i);
546  }
547  }
548 
549  if (verify && !consumed_every_bit) {
550  log_printf("Relocation overflow inst=%08lx mask=%08lx val=%08lx\n",
551  inst, mask, (unsigned long)old_val);
552  fail(__LINE__);
553  return;
554  }
555 
556  if (debug) log_printf("Relocated instruction:\n %08lx\n", inst);
557  *((uint32_t *)addr) = inst;
558  }

847  encoding_table[(triple >> (1*6)) & 0x3f],
848  encoding_table[(triple >> (0*6)) & 0x3f]);
849  }
850  log_printf("\nEND BASE64\n");
851  }
