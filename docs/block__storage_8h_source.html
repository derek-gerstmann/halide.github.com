200void BlockStorage::resize(void *user_context, size_t entry_count, bool realloc) {
201 size_t current_size = capacity;
202 size_t requested_size = entry_count;
203 size_t minimum_size = config.minimum_capacity;
204 size_t actual_size = current_size;
205 count = requested_size;
206
207 // increase capacity upto 1.5x existing (or at least min_capacity)
208 if (requested_size > current_size) {
209 actual_size = max(requested_size, max(current_size * 3 / 2, minimum_size));
210 } else if (!realloc) {
211 return;
212 }
213
214#if DEBUG
215 debug(user_context) << "BlockStorage: Resize ("
216 << "requested_size=" << (int32_t)requested_size << " "
217 << "current_size=" << (int32_t)current_size << " "
218 << "minimum_size=" << (int32_t)minimum_size << " "
219 << "actual_size=" << (int32_t)actual_size << " "
220 << "entry_size=" << (int32_t)config.entry_size << " "
221 << "realloc=" << (realloc ? "true" : "false") << ")...\n";
222#endif
223
224 allocate(user_context, actual_size);

250 halide_debug_assert(user_context, index < count);
251 const size_t last_index = size();
252 if (index < (last_index - entry_count)) {
253 size_t dst_offset = index * config.entry_size;
254 size_t src_offset = (index + entry_count) * config.entry_size;

296void BlockStorage::insert(void *user_context, size_t index, const void *array, size_t array_size) {
297 halide_debug_assert(user_context, index <= count);
298 const size_t last_index = size();
299 resize(user_context, last_index + array_size);
300 if (index < last_index) {
301 size_t src_offset = index * config.entry_size;
302 size_t dst_offset = (index + array_size) * config.entry_size;
303 size_t bytes = (last_index - index) * config.entry_size;
304 void *src_ptr = offset_address(ptr, src_offset);
305 void *dst_ptr = offset_address(ptr, dst_offset);
306 memmove(dst_ptr, src_ptr, bytes);
307 }
308 replace(user_context, index, array, array_size);
309}

372void BlockStorage::allocate(void *user_context, size_t new_capacity) {
373 if (new_capacity != capacity) {
374 halide_abort_if_false(user_context, allocator.allocate != nullptr);
375 size_t requested_bytes = new_capacity * config.entry_size;
376 size_t block_size = max(config.block_size, config.entry_size);
377 size_t block_count = (requested_bytes / block_size);
378 block_count += (requested_bytes % block_size) ? 1 : 0;
379 size_t alloc_size = block_count * block_size;
380#if DEBUG
381 debug(0) << "BlockStorage: Allocating ("
382 << "requested_bytes=" << (int32_t)requested_bytes << " "
383 << "block_size=" << (int32_t)block_size << " "
384 << "block_count=" << (int32_t)block_count << " "
385 << "alloc_size=" << (int32_t)alloc_size << ") ...\n";
386#endif
387 void *new_ptr = alloc_size ? allocator.allocate(user_context, alloc_size) : nullptr;
388 if (count != 0 && ptr != nullptr && new_ptr != nullptr) {
389 memcpy(new_ptr, ptr, count * config.entry_size);
390 }
391 if (ptr != nullptr) {
392 halide_abort_if_false(user_context, allocator.deallocate != nullptr);
393 allocator.deallocate(user_context, ptr);
394 }
395 capacity = new_capacity;
396 ptr = new_ptr;
397 }
