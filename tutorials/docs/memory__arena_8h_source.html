154void *MemoryArena::reserve(void *user_context, bool initialize) {
155 // Scan blocks for a free entry
156 for (size_t i = blocks.size(); i--;) {
157 Block *block = lookup_block(user_context, i);
158 halide_abort_if_false(user_context, block != nullptr);
159 if (block->free_index != invalid_entry) {
160 return create_entry(user_context, block, block->free_index);
161 }
162 }
163
164 if (config.maximum_block_count && (blocks.size() >= config.maximum_block_count)) {
165 halide_error(user_context, "MemoryArena: Failed to reserve new entry! Maxmimum blocks reached!\n");
166 return nullptr;
167 }
168
169 // All blocks full ... create a new one
170 uint32_t index = 0;
171 Block *block = create_block(user_context);
172 void *entry_ptr = create_entry(user_context, block, index);
173
174 // Optionally clear the allocation if requested
175 if (initialize) {
176 memset(entry_ptr, 0, config.entry_size);
177 }
178 return entry_ptr;
179}

181void MemoryArena::reclaim(void *user_context, void *entry_ptr) {
182 for (size_t i = blocks.size(); i--;) {
183 Block *block = lookup_block(user_context, i);
184 halide_abort_if_false(user_context, block != nullptr);
185
186 // is entry_ptr in the address range of this block.
187 uint8_t *offset_ptr = static_cast<uint8_t *>(entry_ptr);
188 uint8_t *base_ptr = static_cast<uint8_t *>(block->entries);
189 uint8_t *end_ptr = static_cast<uint8_t *>(offset_address(block->entries, block->capacity * config.entry_size));
190 if ((entry_ptr >= base_ptr) && (entry_ptr < end_ptr)) {
191 const uint32_t offset = static_cast<uint32_t>(offset_ptr - base_ptr);
192 const uint32_t index = offset / config.entry_size;
193 destroy_entry(user_context, block, index);
194 return;
195 }
196 }
197 halide_error(user_context, "MemoryArena: Pointer address doesn't belong to this memory pool!\n");
198}

200typename MemoryArena::Block *MemoryArena::create_block(void *user_context) {
201 // resize capacity starting with initial up to 1.5 last capacity
202 uint32_t new_capacity = config.minimum_block_capacity;
203 if (!blocks.empty()) {
204 const Block *last_block = static_cast<Block *>(blocks.back());
205 new_capacity = (last_block->capacity * 3 / 2);
206 }
207
208 halide_abort_if_false(user_context, current_allocator().allocate != nullptr);
209 void *new_entries = current_allocator().allocate(user_context, config.entry_size * new_capacity);
210 memset(new_entries, 0, config.entry_size * new_capacity);
211
212 uint32_t *new_indices = (uint32_t *)current_allocator().allocate(user_context, sizeof(uint32_t) * new_capacity);
213 AllocationStatus *new_status = (AllocationStatus *)current_allocator().allocate(user_context, sizeof(AllocationStatus) * new_capacity);
214
215 for (uint32_t i = 0; i < new_capacity - 1; ++i) {
216 new_indices[i] = i + 1; // singly-linked list of all free entries in the block
217 new_status[i] = AllocationStatus::Available; // usage status
218 }
219
220 new_indices[new_capacity - 1] = invalid_entry;
221 new_status[new_capacity - 1] = AllocationStatus::InvalidStatus;
222
223 const Block new_block = {new_entries, new_indices, new_status, new_capacity, 0};
224 blocks.append(user_context, &new_block);
225 return static_cast<Block *>(blocks.back());
226}

228void MemoryArena::destroy_block(void *user_context, Block *block) {
229 halide_abort_if_false(user_context, block != nullptr);
230 if (block->entries != nullptr) {
231 halide_abort_if_false(user_context, current_allocator().deallocate != nullptr);
232 current_allocator().deallocate(user_context, block->entries);
233 current_allocator().deallocate(user_context, block->indices);
234 current_allocator().deallocate(user_context, block->status);
235 block->entries = nullptr;
236 block->indices = nullptr;
237 block->status = nullptr;
238 }

241bool MemoryArena::collect_block(void *user_context, Block *block) {
242 halide_abort_if_false(user_context, block != nullptr);
243 if (block->entries != nullptr) {
244 bool can_collect = true;
245 for (size_t i = block->capacity; i--;) {
246 if (block->status[i] == AllocationStatus::InUse) {
247 can_collect = false;
248 break;
249 }
250 }
251 if (can_collect) {
252 destroy_block(user_context, block);
253 return true;
254 }
255 }
256 return false;
257}
