11struct StringUtils {
12 static bool is_empty(const char *str) {
13 if (str == nullptr) {
14 return true;
15 }
16 if (str[0] == '\0') {
17 return true;
18 }
19 return false;
20 }

22 // count the number of delimited string tokens
23 static size_t count_tokens(const char *str, const char *delim) {
24 if (StringUtils::is_empty(str)) {
25 return 0;
26 }
27 if (StringUtils::is_empty(delim)) {
28 return 1;
29 } // no delim ... string is one token
30
31 size_t count = 0;
32 const char *ptr = str;
33 size_t delim_length = strlen(delim);
34 while (!StringUtils::is_empty(ptr)) {
35 const char *next_delim = strstr(ptr, delim);
36 ptr = (next_delim != nullptr) ? (next_delim + delim_length) : nullptr;
37 ++count;
38 }
39 return count;
40 }
41
42 // retuns true if s1 contains s2 (within n characters)
43 static bool contains(const char *s1, const char *s2, size_t n) {
44 if (is_empty(s2)) {
45 return true;
46 } // s2 is empty ... return true to match strstr
47 char starts_with = *s2;
48 for (size_t length = strlen(s2); length <= n; n--, s1++) {
49 if (*s1 == starts_with) {
50 for (size_t i = 1; i <= length; i++) {
51 if (i == length) {
52 return true;
53 }
54 if (s1[i] != s2[i]) {
55 break;
56 }
57 }
58 }
59 }
60 return false;
61 }
62
63 static size_t count_length(const char *str, size_t max_chars) {
64 const char *ptr = str;
65 while (!StringUtils::is_empty(ptr) && ((size_t(ptr - str)) < max_chars)) {
66 ++ptr;
67 }
68 return size_t(ptr - str);
69 }
70};
