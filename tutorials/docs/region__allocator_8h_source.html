13/** Allocator class interface for sub-allocating a contiguous
14 * memory block into smaller regions of memory. This class only
15 * manages the address creation for the regions -- allocation
16 * callback functions are used to request the memory from the
17 * necessary system or API calls. This class is intended to be
18 * used inside of a higher level memory management class that
19 * provides thread safety, policy management and API
20 * integration for a specific runtime API (eg Vulkan, OpenCL, etc)
21 */

24 // disable copy constructors and assignment

26 RegionAllocator &operator=(const RegionAllocator &) = delete;
27
28 // disable non-factory based construction
29 RegionAllocator() = delete;
30 ~RegionAllocator() = delete;

132MemoryRegion *RegionAllocator::reserve(void *user_context, const MemoryRequest &request) {
133 halide_abort_if_false(user_context, request.size > 0);
134 size_t remaining = block->memory.size - block->reserved;
135 if (remaining < request.size) {
136#ifdef DEBUG_RUNTIME
137 debug(user_context) << "RegionAllocator: Unable to reserve more memory from block "
138 << "-- requested size (" << (int32_t)(request.size) << " bytes) "
139 << "greater than available (" << (int32_t)(remaining) << " bytes)!\n";
140#endif
141 return nullptr;
142 }
143
144 BlockRegion *block_region = find_block_region(user_context, request);
145 if (block_region == nullptr) {
146#ifdef DEBUG_RUNTIME
147 debug(user_context) << "RegionAllocator: Failed to locate region for requested size ("
148 << (int32_t)(request.size) << " bytes)!\n";
149#endif
150 return nullptr;
151 }
152
153 if (can_split(block_region, request.size)) {
154#ifdef DEBUG_RUNTIME
155 debug(user_context) << "RegionAllocator: Splitting region of size ( " << (int32_t)(block_region->memory.size) << ") "
156 << "to accomodate requested size (" << (int32_t)(request.size) << " bytes)!\n";
157#endif
158 split_block_region(user_context, block_region, request.size, request.alignment);
159 }
160
161 alloc_block_region(user_context, block_region);
162 return reinterpret_cast<MemoryRegion *>(block_region);
163}

182BlockRegion *RegionAllocator::find_block_region(void *user_context, const MemoryRequest &request) {
183 BlockRegion *result = nullptr;
184 for (BlockRegion *block_region = block->regions; block_region != nullptr; block_region = block_region->next_ptr) {
185
186 if (block_region->status != AllocationStatus::Available) {
187 continue;
188 }
189
190 // skip incompatible block regions for this request
191 if (!is_compatible_block_region(block_region, request.properties)) {
192 continue;
193 }
194
195 // is the requested size larger than the current region?
196 if (request.size > block_region->memory.size) {
197 continue;
198 }
199
200 size_t actual_size = aligned_size(block_region->memory.offset, request.size, request.alignment);
201
202 // is the adjusted size larger than the current region?
203 if (actual_size > block_region->memory.size) {
204 continue;
205 }
206
207 // will the adjusted size fit within the remaining unallocated space?
208 if ((actual_size + block->reserved) < block->memory.size) {
209 result = block_region; // best-fit!
210 break;
211 }
212 }
213 return result;
214}

216bool RegionAllocator::can_coalesce(BlockRegion *block_region) {
217 if (block_region == nullptr) {
218 return false;
219 }
220 if (block_region->prev_ptr && (block_region->prev_ptr->status == AllocationStatus::Available)) {
221 return true;
222 }

224 return true;
225 }

229BlockRegion *RegionAllocator::coalesce_block_regions(void *user_context, BlockRegion *block_region) {
230 if (block_region->prev_ptr && (block_region->prev_ptr->status == AllocationStatus::Available)) {
231 BlockRegion *prev_region = block_region->prev_ptr;
232
233#ifdef DEBUG_RUNTIME
234 debug(user_context) << "RegionAllocator: Coalescing "
235 << "previous region (offset=" << (int32_t)prev_region->memory.offset << " size=" << (int32_t)(prev_region->memory.size) << " bytes) "
236 << "into current region (offset=" << (int32_t)block_region->memory.offset << " size=" << (int32_t)(block_region->memory.size) << " bytes)\n!";
237#endif
238
239 prev_region->next_ptr = block_region->next_ptr;
240 if (block_region->next_ptr) {
241 block_region->next_ptr->prev_ptr = prev_region;
242 }
243 prev_region->memory.size += block_region->memory.size;
244 destroy_block_region(user_context, block_region);
245 block_region = prev_region;
246 }
247
248 if (block_region->next_ptr && (block_region->next_ptr->status == AllocationStatus::Available)) {
249 BlockRegion *next_region = block_region->next_ptr;
250
251#ifdef DEBUG_RUNTIME
252 debug(user_context) << "RegionAllocator: Coalescing "
253 << "next region (offset=" << (int32_t)next_region->memory.offset << " size=" << (int32_t)(next_region->memory.size) << " bytes) "
254 << "into current region (offset=" << (int32_t)block_region->memory.offset << " size=" << (int32_t)(block_region->memory.size) << " bytes)!\n";
255#endif
256
257 if (next_region->next_ptr) {
258 next_region->next_ptr->prev_ptr = block_region;
259 }
260 block_region->next_ptr = next_region->next_ptr;
261 block_region->memory.size += next_region->memory.size;
262 destroy_block_region(user_context, next_region);
263 }
264
265 return block_region;
266}

272BlockRegion *RegionAllocator::split_block_region(void *user_context, BlockRegion *block_region, size_t size, size_t alignment) {
273 size_t adjusted_size = aligned_size(block_region->memory.offset, size, alignment);
274 size_t adjusted_offset = aligned_offset(block_region->memory.offset, alignment);
275
276 size_t empty_offset = adjusted_offset + size;
277 size_t empty_size = block_region->memory.size - adjusted_size;
278
279#ifdef DEBUG_RUNTIME
280 debug(user_context) << "RegionAllocator: Splitting "
281 << "current region (offset=" << (int32_t)block_region->memory.offset << " size=" << (int32_t)(block_region->memory.size) << " bytes) "
282 << "to create empty region (offset=" << (int32_t)empty_offset << " size=" << (int32_t)(empty_size) << " bytes)!\n";
283#endif
284
285 BlockRegion *next_region = block_region->next_ptr;
286 BlockRegion *empty_region = create_block_region(user_context,
287 block_region->memory.properties,
288 empty_offset, empty_size,
289 block_region->memory.dedicated);
290 halide_abort_if_false(user_context, empty_region != nullptr);
291
292 empty_region->next_ptr = next_region;
293 if (next_region) {
294 next_region->prev_ptr = empty_region;
295 }
296 block_region->next_ptr = empty_region;
297 block_region->memory.size = size;
298 return empty_region;
299}

436 if (properties.caching != MemoryCaching::DefaultCaching) {
437 if (properties.caching != block_region->memory.properties.caching) {
438 return false;

443 if (properties.visibility != block_region->memory.properties.visibility) {

449 if (properties.usage != block_region->memory.properties.usage) {
450 return false;
