14/** Allocator class interface for managing large contiguous blocks
15 * of memory, which are then sub-allocated into smaller regions of
16 * memory. This class only manages the address creation for the
17 * regions -- allocation callback functions are used to request the
18 * memory from the necessary system or API calls. This class is
19 * intended to be used inside of a higher level memory management
20 * class that provides thread safety, policy management and API
21 * integration for a specific runtime API (eg Vulkan, OpenCL, etc)
22 */

24public:
25 // disable copy constructors and assignment

27 BlockAllocator &operator=(const BlockAllocator &) = delete;
28
29 // disable non-factory based construction

31 ~BlockAllocator() = delete;

144MemoryRegion *BlockAllocator::reserve(void *user_context, const MemoryRequest &request) {
145#ifdef DEBUG_RUNTIME
146 debug(user_context) << "BlockAllocator: Reserve ("
147 << "user_context=" << (void *)(user_context) << " "
148 << "offset=" << (uint32_t)request.offset << " "
149 << "size=" << (uint32_t)request.size << " "
150 << "dedicated=" << (request.dedicated ? "true" : "false") << " "
151 << "usage=" << halide_memory_usage_name(request.properties.usage) << " "
152 << "caching=" << halide_memory_caching_name(request.properties.caching) << " "
153 << "visibility=" << halide_memory_visibility_name(request.properties.visibility) << ") ...\n";
154#endif
155 BlockEntry *block_entry = reserve_block_entry(user_context, request.properties, request.size, request.dedicated);
156 if (block_entry == nullptr) {
157 debug(user_context) << "BlockAllocator: Failed to allocate new empty block of requested size ("
158 << (int32_t)(request.size) << " bytes)!\n";
159 return nullptr;
160 }
161
162 BlockResource *block = static_cast<BlockResource *>(block_entry->value);
163 halide_abort_if_false(user_context, block != nullptr);
164 halide_abort_if_false(user_context, block->allocator != nullptr);
165
166 MemoryRegion *result = reserve_memory_region(user_context, block->allocator, request);
167 if (result == nullptr) {
168
169 // Unable to reserve region in an existing block ... create a new block and try again.
170 size_t actual_size = constrain_requested_size(request.size);
171 block_entry = create_block_entry(user_context, request.properties, actual_size, request.dedicated);
172 if (block_entry == nullptr) {
173 debug(user_context) << "BlockAllocator: Out of memory! Failed to allocate empty block of size ("
174 << (int32_t)(actual_size) << " bytes)!\n";
175 return nullptr;
176 }
177
178 block = static_cast<BlockResource *>(block_entry->value);
179 if (block->allocator == nullptr) {
180 block->allocator = create_region_allocator(user_context, block);
181 }
182
183 result = reserve_memory_region(user_context, block->allocator, request);
184 }
185 return result;
186}

197bool BlockAllocator::collect(void *user_context) {
198 bool result = false;
199 BlockEntry *block_entry = block_list.back();
200 while (block_entry != nullptr) {
201 BlockEntry *prev_entry = block_entry->prev_ptr;
202
203 const BlockResource *block = static_cast<BlockResource *>(block_entry->value);
204 if (block->allocator == nullptr) {
205 block_entry = prev_entry;
206 continue;
207 }
208
209 block->allocator->collect(user_context);
210 if (block->reserved == 0) {
211 destroy_block_entry(user_context, block_entry);
212 result = true;
213 }
214
215 block_entry = prev_entry;
216 }
217 return result;
218}

239MemoryRegion *BlockAllocator::reserve_memory_region(void *user_context, RegionAllocator *allocator, const MemoryRequest &request) {
240 MemoryRegion *result = allocator->reserve(user_context, request);
241 if (result == nullptr) {
242#ifdef DEBUG_RUNTIME
243 debug(user_context) << "BlockAllocator: Failed to allocate region of size ("
244 << (int32_t)(request.size) << " bytes)!\n";
245#endif
246 // allocator has enough free space, but not enough contiguous space
247 // -- collect and try to reallocate
248 if (allocator->collect(user_context)) {
249 result = allocator->reserve(user_context, request);
250 }
251 }
252 return result;
253}

255BlockAllocator::BlockEntry *
256BlockAllocator::find_block_entry(void *user_context, const MemoryProperties &properties, size_t size, bool dedicated) {
257 BlockEntry *block_entry = nullptr;
258 for (block_entry = block_list.front(); block_entry != nullptr; block_entry = block_entry->next_ptr) {
259
260 const BlockResource *block = static_cast<BlockResource *>(block_entry->value);
261 if (!is_compatible_block(block, properties)) {
262 continue;
263 }
264
265 // skip blocks that can't be dedicated to a single allocation
266 if (dedicated && (block->reserved > 0)) {
267 continue;
268 }
269
270 // skip dedicated blocks that are already allocated
271 if (block->memory.dedicated && (block->reserved > 0)) {
272 continue;
273 }
274
275 size_t available = (block->memory.size - block->reserved);
276 if (available >= size) {
277#ifdef DEBUG_RUNTIME
278 debug(user_context) << "BlockAllocator: find_block_entry (FOUND) ("
279 << "user_context=" << (void *)(user_context) << " "
280 << "block_entry=" << (void *)(block_entry) << " "
281 << "size=" << (uint32_t)size << " "
282 << "dedicated=" << (dedicated ? "true" : "false") << " "
283 << "usage=" << halide_memory_usage_name(properties.usage) << " "
284 << "caching=" << halide_memory_caching_name(properties.caching) << " "
285 << "visibility=" << halide_memory_visibility_name(properties.visibility) << ") ...\n";
286#endif
287 break;
288 }
289 }
290
291 return block_entry;
292}

342BlockAllocator::BlockEntry *
343BlockAllocator::create_block_entry(void *user_context, const MemoryProperties &properties, size_t size, bool dedicated) {
344 if (config.maximum_block_count && (block_count() >= config.maximum_block_count)) {
345 debug(user_context) << "BlockAllocator: No free blocks found! Maximum block count reached ("
346 << (int32_t)(config.maximum_block_count) << ")!\n";
347 return nullptr;
348 }
349
350 BlockEntry *block_entry = block_list.append(user_context);
351 if (block_entry == nullptr) {
352 debug(user_context) << "BlockAllocator: Failed to allocate new block entry!\n";
353 return nullptr;
354 }
355
356#ifdef DEBUG_RUNTIME
357 debug(user_context) << "BlockAllocator: Creating block entry ("
358 << "block_entry=" << (void *)(block_entry) << " "
359 << "block=" << (void *)(block_entry->value) << " "
360 << "allocator=" << (void *)(allocators.block.allocate) << ")...\n";
361#endif
362
363 BlockResource *block = static_cast<BlockResource *>(block_entry->value);
364 block->memory.size = size;
365 block->memory.properties = properties;
366 block->memory.dedicated = dedicated;
367 block->reserved = 0;
368 block->allocator = create_region_allocator(user_context, block);
369 alloc_memory_block(user_context, block);
370 return block_entry;
371}

437bool BlockAllocator::is_compatible_block(const BlockResource *block, const MemoryProperties &properties) const {
438 if (properties.caching != MemoryCaching::DefaultCaching) {
439 if (properties.caching != block->memory.properties.caching) {
440 return false;
441 }
442 }
443
444 if (properties.visibility != MemoryVisibility::DefaultVisibility) {
445 if (properties.visibility != block->memory.properties.visibility) {
446 return false;
447 }
448 }
449
450 if (properties.usage != MemoryUsage::DefaultUsage) {
451 if (properties.usage != block->memory.properties.usage) {
452 return false;
453 }
454 }
455
456 return true;
457}
