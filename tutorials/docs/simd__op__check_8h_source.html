127 // Check if pattern p matches str, allowing for wildcards (*).
128 bool wildcard_match(const char *p, const char *str) const {
129 // Match all non-wildcard characters.
130 while (*p && *str && *p == *str && *p != '*') {
131 str++;
132 p++;
133 }
134
135 if (!*p) {
136 return *str == 0;
137 } else if (*p == '*') {
138 p++;
139 do {
140 if (wildcard_match(p, str)) {
141 return true;
142 }
143 } while (*str++);
144 } else if (*p == ' ') { // ignore whitespace in pattern
145 p++;
146 if (wildcard_match(p, str)) {
147 return true;
148 }
149 } else if (*str == ' ') { // ignore whitespace in string
150 str++;
151 if (wildcard_match(p, str)) {
152 return true;
153 }
154 }
155 return !*p;
156 }

170 class HasInlineReduction : public Internal::IRVisitor {
171 using Internal::IRVisitor::visit;
172 void visit(const Internal::Call *op) override {
173 if (op->call_type == Internal::Call::Halide) {
174 Internal::Function f(op->func);
175 if (f.has_update_definition()) {
176 inline_reduction = f;
177 result = true;
178 }
179 }
180 IRVisitor::visit(op);
181 }
182
183 public:
184 Internal::Function inline_reduction;
185 bool result = false;
186 } has_inline_reduction;
187 e.accept(&has_inline_reduction);
188
189 // Define a vectorized Halide::Func that uses the pattern.
190 Halide::Func f(name);
191 f(x, y) = e;
192 f.bound(x, 0, W).vectorize(x, vector_width);
193 f.compute_root();
194
195 // Include a scalar version
196 Halide::Func f_scalar("scalar_" + name);
197 f_scalar(x, y) = e;
198
199 if (has_inline_reduction.result) {
200 // If there's an inline reduction, we want to vectorize it
201 // over the RVar.
202 Var xo, xi;
203 RVar rxi;
204 Func g{has_inline_reduction.inline_reduction};
205
206 // Do the reduction separately in f_scalar
207 g.clone_in(f_scalar);
208
209 g.compute_at(f, x)
210 .update()
211 .split(x, xo, xi, vector_width)
212 .atomic(true)
213 .vectorize(g.rvars()[0])
214 .vectorize(xi);
215 }
216
217 // The output to the pipeline is the maximum absolute difference as a double.
218 RDom r_check(0, W, 0, H);
219 Halide::Func error("error_" + name);
220 error() = Halide::cast<double>(maximum(absd(f(r_check.x, r_check.y), f_scalar(r_check.x, r_check.y))));
221
222 setup_images();
223 compile_and_check(error, op, name, vector_width, error_msg);
224
225 bool can_run_the_code = can_run_code();
226 if (can_run_the_code) {
227 Target run_target = target
228 .without_feature(Target::NoRuntime)
229 .without_feature(Target::NoAsserts)
230 .without_feature(Target::NoBoundsQuery);
231
232 error.infer_input_bounds({}, run_target);
233 // Fill the inputs with noise
234 for (auto p : image_params) {
235 Halide::Buffer<> buf = p.get();
236 if (!buf.defined()) continue;
237 assert(buf.data());
238 Type t = buf.type();
239 // For floats/doubles, we only use values that aren't
240 // subject to rounding error that may differ between
241 // vectorized and non-vectorized versions
242 if (t == Float(32)) {
243 buf.as<float>().for_each_value([&](float &f) { f = (rng() & 0xfff) / 8.0f - 0xff; });
244 } else if (t == Float(64)) {
245 buf.as<double>().for_each_value([&](double &f) { f = (rng() & 0xfff) / 8.0 - 0xff; });
246 } else if (t == Float(16)) {
247 buf.as<float16_t>().for_each_value([&](float16_t &f) { f = float16_t((rng() & 0xff) / 8.0f - 0xf); });
248 } else {
249 // Random bits is fine
250 for (uint32_t *ptr = (uint32_t *)buf.data();
251 ptr != (uint32_t *)buf.data() + buf.size_in_bytes() / 4;
252 ptr++) {
253 // Never use the top four bits, to avoid
254 // signed integer overflow.
255 *ptr = ((uint32_t)rng()) & 0x0fffffff;
256 }
257 }
258 }
259 Realization r = error.realize();
260 double e = Buffer<double>(r[0])();
261 // Use a very loose tolerance for floating point tests. The
262 // kinds of bugs we're looking for are codegen bugs that
263 // return the wrong value entirely, not floating point
264 // accuracy differences between vectors and scalars.
265 if (e > 0.001) {
266 error_msg << "The vector and scalar versions of " << name << " disagree. Maximum error: " << e << "\n";
267
268 std::string error_filename = output_directory + "error_" + name + ".s";
269 error.compile_to_assembly(error_filename, arg_types, target);
270
271 std::ifstream error_file;
272 error_file.open(error_filename);
273
274 error_msg << "Error assembly: \n";
275 std::string line;
276 while (getline(error_file, line)) {
277 error_msg << line << "\n";
278 }
279
280 error_file.close();
281 }
282 }
283
284 return {op, error_msg.str()};
285 }
